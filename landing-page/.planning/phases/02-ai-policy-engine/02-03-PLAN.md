   ```typescript
   export function updateItemStatus(itemId: number, status: ItemStatus): void {
     db.prepare('UPDATE approval_items SET status = ?, updated_at = CURRENT_TIMESTAMP WHERE id = ?')
       .run(status, itemId);
   }

   export function createTask(task: {
     approval_item_id: number;
     owner: Lane;
     title: string;
     description: string | null;
     due_date: string | null;
   }): number {
     const stmt = db.prepare(`
       INSERT INTO tasks (approval_item_id, owner, title, description, due_date)
       VALUES (?, ?, ?, ?, ?)
     `);
     const result = stmt.run(
       task.approval_item_id,
       task.owner,
       task.title,
       task.description,
       task.due_date
     );
     return result.lastInsertRowid as number;
   }

   export function logAuditEntry(entry: {
     approval_item_id: number;
     action: string;
     actor: string;
     details: Record<string, any>;
   }): number {
     const stmt = db.prepare(`
       INSERT INTO audit_log (approval_item_id, action, actor, details)
       VALUES (?, ?, ?, ?)
     `);
     const result = stmt.run(
       entry.approval_item_id,
       entry.action,
       entry.actor,
       JSON.stringify(entry.details)
     );
     return result.lastInsertRowid as number;
   }

   export function getAuditLog(itemId: number): AuditLogEntry[] {
     const rows = db.prepare(`
       SELECT * FROM audit_log WHERE approval_item_id = ? ORDER BY created_at DESC
     `).all(itemId) as any[];

     return rows.map(row => ({
       ...row,
       details: row.details ? JSON.parse(row.details) : null,
     }));
   }
   ```

2. Create app/api/items/[id]/decision/route.ts:
   ```typescript
   import { NextRequest, NextResponse } from 'next/server';
   import {
     getItemById,
     getLatestAnalysis,
     updateItemStatus,
     createTask,
     logAuditEntry,
   } from '@/lib/queries';
   import type { ItemStatus, Lane, SuggestedTask } from '@/lib/types';

   interface DecisionRequest {
     decision: 'approve' | 'needs_changes' | 'reject' | 'escalate';
     feedback?: string;
     create_tasks?: boolean;
   }

   const DECISION_TO_STATUS: Record<string, ItemStatus> = {
     approve: 'approved',
     needs_changes: 'needs_changes',
     reject: 'rejected',
     escalate: 'escalated',
   };

   export async function POST(
     request: NextRequest,
     { params }: { params: Promise<{ id: string }> }
   ) {
     try {
       const { id } = await params;
       const itemId = parseInt(id, 10);

       if (isNaN(itemId)) {
         return NextResponse.json({ error: 'Invalid item ID' }, { status: 400 });
       }

       const item = getItemById(itemId);
       if (!item) {
         return NextResponse.json({ error: 'Item not found' }, { status: 404 });
       }

       const body: DecisionRequest = await request.json();
       const { decision, feedback, create_tasks } = body;

       if (!decision || !DECISION_TO_STATUS[decision]) {
         return NextResponse.json({ error: 'Invalid decision' }, { status: 400 });
       }

       // Update item status
       const newStatus = DECISION_TO_STATUS[decision];
       updateItemStatus(itemId, newStatus);

       // Get analysis for task creation
       const analysis = getLatestAnalysis(itemId);

       // Create tasks if requested and analysis has suggestions
       const createdTasks: number[] = [];
       if (create_tasks && analysis?.suggested_tasks?.length) {
         for (const task of analysis.suggested_tasks) {
           const dueDate = task.due_days
             ? new Date(Date.now() + task.due_days * 24 * 60 * 60 * 1000).toISOString().split('T')[0]
             : null;

           const taskId = createTask({
             approval_item_id: itemId,
             owner: task.owner,
             title: task.title,
             description: task.description,
             due_date: dueDate,
           });
           createdTasks.push(taskId);
         }
       }

       // Log audit entry
       const actor = item.lane === 'xixi' ? 'Xixi' : 'Allen'; // Demo: derive from lane
       logAuditEntry({
         approval_item_id: itemId,
         action: `decision_${decision}`,
         actor,
         details: {
           previous_status: item.status,
           new_status: newStatus,
           feedback: feedback || null,
           tasks_created: createdTasks.length,
           analysis_confidence: analysis?.confidence,
           analysis_recommendation: analysis?.recommendation,
         },
       });

       return NextResponse.json({
         success: true,
         item_id: itemId,
         new_status: newStatus,
         tasks_created: createdTasks,
       });
     } catch (error) {
       console.error('Decision error:', error);
       return NextResponse.json(
         { error: 'Decision failed', details: error instanceof Error ? error.message : 'Unknown error' },
         { status: 500 }
       );
     }
   }
   ```

2. Create app/components/AuditTimeline.tsx:
   ```typescript
   'use client';

   import { useEffect, useState } from 'react';
   import type { AuditLogEntry } from '@/lib/types';

   interface AuditTimelineProps {
     itemId: number;
   }

   export default function AuditTimeline({ itemId }: AuditTimelineProps) {
     const [entries, setEntries] = useState<AuditLogEntry[]>([]);
     const [loading, setLoading] = useState(true);

     useEffect(() => {
       fetch(`/api/items/${itemId}/audit`)
         .then(res => res.json())
         .then(data => {
           setEntries(data.entries || []);
           setLoading(false);
         })
         .catch(() => setLoading(false));
     }, [itemId]);

     if (loading) {
       return <div className="animate-pulse">Loading audit log...</div>;
     }

     if (entries.length === 0) {
       return (
         <div className="text-sm text-gray-500">
           No audit entries yet
         </div>
       );
     }

     const formatAction = (action: string) => {
       const actionMap: Record<string, { label: string; color: string }> = {
         'decision_approve': { label: 'Approved', color: 'text-green-600' },
         'decision_needs_changes': { label: 'Needs Changes', color: 'text-yellow-600' },
         'decision_reject': { label: 'Rejected', color: 'text-red-600' },
         'decision_escalate': { label: 'Escalated', color: 'text-purple-600' },
         'analysis_generated': { label: 'Analysis Generated', color: 'text-blue-600' },
       };
       return actionMap[action] || { label: action, color: 'text-gray-600' };
     };

     return (
       <div className="space-y-3">
         <h3 className="text-sm font-medium text-gray-700">Audit Log</h3>
         <div className="space-y-2">
           {entries.map((entry) => {
             const { label, color } = formatAction(entry.action);
             return (
               <div key={entry.id} className="flex items-start gap-3 text-sm">
                 <div className="flex-shrink-0 w-2 h-2 mt-1.5 rounded-full bg-gray-300" />
                 <div className="flex-1">
                   <div className="flex items-center gap-2">
                     <span className={`font-medium ${color}`}>{label}</span>
                     <span className="text-gray-400">by {entry.actor}</span>
                   </div>
                   <div className="text-xs text-gray-400">
                     {new Date(entry.created_at).toLocaleString()}
                   </div>
                   {entry.details?.feedback && (
                     <div className="mt-1 text-gray-600">
                       "{entry.details.feedback}"
                     </div>
                   )}
                   {entry.details?.tasks_created > 0 && (
                     <div className="mt-1 text-gray-500">
                       {entry.details.tasks_created} task(s) created
                     </div>
                   )}
                 </div>
               </div>
             );
           })}
         </div>
       </div>
     );
   }
   ```
