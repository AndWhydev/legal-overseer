```typescript
export interface AgentRequest {
  message: string;
  channel: 'whatsapp' | 'email' | 'voice' | 'sms';
  sender: {
    type: 'customer' | 'xixi' | 'allen';
    email?: string;
    phone?: string;
    name?: string;
  };
  context?: {
    order_number?: string;
    previous_messages?: string[];
  };
}

export interface AgentAction {
  type: string;
  params: Record<string, unknown>;
  result: unknown;
}

export interface AgentResponse {
  message: string;           // Response to send back
  actions_taken: AgentAction[];
  reasoning: string;         // Why agent chose this
  confidence: number;        // 0-100
  session_id: string;        // For audit trail
  should_escalate: boolean;  // True if <50% confidence
}
```

<task type="auto">
  <name>Task 3: Create agent API endpoint</name>
  <files>app/api/agent/route.ts</files>

Create POST /api/agent that:

1. **Parse request:**
   - Validate AgentRequest body
   - Generate session_id (UUID)

2. **Build Claude messages:**
   - System prompt (will be enhanced in plan 02, basic for now)
   - User message with context

3. **Call Claude with tools:**
   ```typescript
   const response = await client.messages.create({
     model: 'claude-sonnet-4-20250514',
     max_tokens: 4096,
     system: systemPrompt,
     tools: agentTools,
     messages: [{ role: 'user', content: userMessage }]
   });
   ```

4. **Handle tool use loop:**
   - If response has tool_use blocks, execute each
   - Send tool_result back to Claude
   - Continue until text response (stop_reason !== 'tool_use')
   - Max 5 iterations to prevent runaway

5. **Build AgentResponse:**
   - Extract message from final text block
   - Collect all actions_taken
   - Extract reasoning (from Claude's thinking)
   - Set confidence (default 75 for now, will be enhanced in plan 02)
   - Set should_escalate based on confidence

6. **Log to agent_actions:**
   - Final action_type='agent_response'
   - Full input/output for audit

7. **Return response**

Handle errors gracefully — don't crash on Claude errors or tool failures.
