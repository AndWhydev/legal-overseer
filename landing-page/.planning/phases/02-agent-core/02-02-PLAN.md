export function checkEscalationTriggers(message: string): string | null {
  const triggers = [
    { pattern: /legal|lawyer|sue|lawsuit/i, reason: 'Legal threat detected' },
    { pattern: /chargeback|dispute|bank/i, reason: 'Chargeback/payment dispute' },
    { pattern: /unsafe|hurt|injured|allergic/i, reason: 'Safety complaint' },
    { pattern: /influencer.*contract|contract.*influencer/i, reason: 'Influencer contract issue' },
  ];

  for (const trigger of triggers) {
    if (trigger.pattern.test(message)) {
      return trigger.reason;
    }
  }
  return null;
}
