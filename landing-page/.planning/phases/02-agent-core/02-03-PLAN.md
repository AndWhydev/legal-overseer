import db from '../db';

export interface AuditLogEntry {
  session_id: string;
  action_type: 'request' | 'tool_call' | 'response' | 'escalation' | 'error';
  input: unknown;
  output?: unknown;
  reasoning?: string;
  confidence?: number;
  success: boolean;
  error_message?: string;
  metadata?: Record<string, unknown>;
}

export function logAuditEntry(entry: AuditLogEntry): number {
  const stmt = db.prepare(`
    INSERT INTO agent_actions (
      session_id, action_type, input, output, reasoning, confidence, success, error_message, created_at
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, datetime('now'))
  `);

  const result = stmt.run(
    entry.session_id,
    entry.action_type,
    JSON.stringify(entry.input),
    entry.output ? JSON.stringify(entry.output) : null,
    entry.reasoning,
    entry.confidence,
    entry.success ? 1 : 0,
    entry.error_message
  );

  return result.lastInsertRowid as number;
}

export function getSessionAuditTrail(sessionId: string): AuditLogEntry[] {
  const rows = db.prepare(`
    SELECT * FROM agent_actions WHERE session_id = ? ORDER BY created_at ASC
  `).all(sessionId);

  return rows.map(row => ({
    ...row,
    input: JSON.parse(row.input),
    output: row.output ? JSON.parse(row.output) : null,
    success: Boolean(row.success),
  }));
}
