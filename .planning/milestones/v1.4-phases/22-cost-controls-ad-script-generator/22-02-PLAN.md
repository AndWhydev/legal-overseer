export type Platform = 'reels' | 'tiktok' | 'shorts' | 'feed'
export type HookType = 'curiosity' | 'problem-agitation' | 'social-proof' | 'direct-offer'

export interface GenerateScriptsParams {
  offerPackageId: string
  platforms: Platform[]
  hookTypes: HookType[]
}

export interface GenerateScriptsResult {
  id?: string
  scripts: AdScript[]
  variations: AdScriptVariation[]
  offerPackageName: string
  generatedAt: string
}

export interface AdScript {
  id?: string
  platform: Platform
  hookType: HookType
  script: string
  duration: number
  shotDescriptions: string[]
  storyboard: StoryboardShot[]
  tone: string
}

export interface AdScriptVariation {
  variantLabel: string
  openingLine: string
  callToAction: string
  tone: 'urgent' | 'casual' | 'professional'
  script: string
}

export interface SavedScriptBatch {
  id: string
  org_id: string
  offer_package_id: string
  offer_name: string
  scripts: AdScript[]
  variations: AdScriptVariation[]
  created_at: string
}

export const adScriptGen = {
  generate: generateScripts,      // (supabase, orgId, params) => GenerateScriptsResult
  generateVariations,             // (supabase, orgId, script, count) => AdScriptVariation[]
  generateStoryboard,             // (script, platform, orgId, supabase) => StoryboardShot[]
  adaptForPlatform,               // (script, fromPlatform, toPlatform) => AdScript
  listBatches: listScriptBatches, // (supabase, orgId, limit?) => SavedScriptBatch[]
  saveBatch: saveScriptBatch,     // (supabase, orgId, batch) => SavedScriptBatch
  tick: runAdScriptGenTick,       // (supabase, orgId) => AdScriptTickResult
  PLATFORM_CONFIGS,               // Record<Platform, PlatformConfig>
  ALL_PLATFORMS,                   // Platform[]
  ALL_HOOK_TYPES,                  // HookType[]
}
