# Plan 43-04: NL Delegation Activation & Revocation

<objective>
Enable natural language activation ("Take Steve off my hands") and revocation ("Stop managing Steve") of entity delegation mandates. Intent detection integrated into the TAOR loop. Revocation is immediate -- cancels in-flight async tasks (Phase 39 semantics) and returns entity to standard routing with zero lag.
</objective>

<must_haves>
- NL intent detection for delegation activation and revocation
- Entity resolution from natural language to entity_id via Memory Palace
- Confirmation flow in chat (BitBit confirms understanding)
- Immediate revocation with in-flight task cancellation
- Integration with TAOR loop decision cycle
</must_haves>

```typescript
import { logger } from '@/lib/core/logger'

export type DelegationIntentType = 'activate' | 'revoke' | null

export interface DelegationIntent {
  type: DelegationIntentType
  entityMention: string  // Raw text mention of the entity
  confidence: number     // 0-1 confidence in intent detection
  rawMessage: string
}

// Activation patterns: user wants BitBit to manage an entity autonomously
const ACTIVATION_PATTERNS: Array<{ pattern: RegExp; confidence: number }> = [
  { pattern: /take\s+(.+?)\s+off\s+my\s+hands/i, confidence: 0.95 },
  { pattern: /manage\s+(.+?)\s+for\s+me/i, confidence: 0.90 },
  { pattern: /handle\s+everything\s+(?:for|with)\s+(.+)/i, confidence: 0.90 },
  { pattern: /put\s+(.+?)\s+on\s+autopilot/i, confidence: 0.95 },
  { pattern: /(.+?)\s+is\s+all\s+yours/i, confidence: 0.85 },
  { pattern: /fully\s+delegate\s+(.+)/i, confidence: 0.95 },
  { pattern: /you\s+(?:handle|manage|run)\s+(.+?)(?:\s+from\s+now\s+on)?$/i, confidence: 0.85 },
]

// Revocation patterns: user wants to take back control
const REVOCATION_PATTERNS: Array<{ pattern: RegExp; confidence: number }> = [
  { pattern: /stop\s+managing\s+(.+)/i, confidence: 0.95 },
  { pattern: /take\s+(.+?)\s+back/i, confidence: 0.85 },
  { pattern: /i'?ll\s+handle\s+(.+?)(?:\s+from\s+now\s+on)?$/i, confidence: 0.85 },
  { pattern: /(.+?)\s+is\s+no\s+longer\s+delegated/i, confidence: 0.95 },
  { pattern: /stop\s+(?:auto|autopilot|autonomous)\s+(?:for\s+)?(.+)/i, confidence: 0.90 },
  { pattern: /revoke\s+delegation\s+(?:for\s+)?(.+)/i, confidence: 0.95 },
  { pattern: /undelegat[e]?\s+(.+)/i, confidence: 0.90 },
]

/**
 * Detect delegation intent from a user message.
 * Returns null intent type if no delegation pattern is found.
 */
export function detectDelegationIntent(message: string): DelegationIntent {
  const trimmed = message.trim()

  // Check activation patterns first
  for (const { pattern, confidence } of ACTIVATION_PATTERNS) {
    const match = trimmed.match(pattern)
    if (match && match[1]) {
      const entityMention = match[1].trim().replace(/[.!?,;:]+$/, '')
      logger.info('[delegation-intent] Activation detected', { entityMention, confidence })
      return { type: 'activate', entityMention, confidence, rawMessage: trimmed }
    }
  }

  // Check revocation patterns
  for (const { pattern, confidence } of REVOCATION_PATTERNS) {
    const match = trimmed.match(pattern)
    if (match && match[1]) {
      const entityMention = match[1].trim().replace(/[.!?,;:]+$/, '')
      logger.info('[delegation-intent] Revocation detected', { entityMention, confidence })
      return { type: 'revoke', entityMention, confidence, rawMessage: trimmed }
    }
  }

  return { type: null, entityMention: '', confidence: 0, rawMessage: trimmed }
}

/**
 * Resolve an entity mention to an entity_id via Memory Palace entity search.
 * Returns null if no matching entity is found.
 */
export async function resolveEntityFromMention(
  supabase: import('@supabase/supabase-js').SupabaseClient,
  orgId: string,
  mention: string,
): Promise<{ id: string; name: string } | null> {
  // Try exact name match first
  const { data: exactMatch } = await supabase
    .from('entity_nodes')
    .select('id, name')
    .eq('org_id', orgId)
    .ilike('name', mention)
    .limit(1)
    .single()

  if (exactMatch) return exactMatch as { id: string; name: string }

  // Try fuzzy match (contains)
  const { data: fuzzyMatches } = await supabase
    .from('entity_nodes')
    .select('id, name')
    .eq('org_id', orgId)
    .ilike('name', `%${mention}%`)
    .limit(3)

  if (fuzzyMatches && fuzzyMatches.length === 1) {
    return fuzzyMatches[0] as { id: string; name: string }
  }

  // Ambiguous or no match
  if (fuzzyMatches && fuzzyMatches.length > 1) {
    logger.warn('[delegation-intent] Ambiguous entity mention', {
      mention,
      candidates: fuzzyMatches.map(m => (m as { name: string }).name),
    })
  }

  return null
}

/**
 * Generate confirmation message for delegation activation.
 */
export function generateActivationConfirmation(entityName: string): string {
  return `Got it. I'll manage ${entityName} autonomously from now on. I'll handle everything — invoicing, follow-ups, scheduling — and send you a morning briefing with what I did. Say "stop managing ${entityName}" anytime to take back control.`
}

/**
 * Generate confirmation message for delegation revocation.
 */
export function generateRevocationConfirmation(entityName: string): string {
  return `Done. I've stopped managing ${entityName} autonomously. Any in-flight tasks have been cancelled (partial work is preserved). ${entityName} is back to standard routing — I'll ask before acting.`
}
```

```typescript
// Check for delegation intent (NL activation/revocation)
const delegationIntent = detectDelegationIntent(userMessage)
if (delegationIntent.type && delegationIntent.confidence >= 0.85) {
  const entity = await resolveEntityFromMention(supabase, orgId, delegationIntent.entityMention)

  if (entity) {
    if (delegationIntent.type === 'activate') {
      await setEntityMandate(supabase, orgId, entity.id, 'infinite_autopilot', 'chat')
      // Return confirmation as assistant response
      yield { type: 'text', text: generateActivationConfirmation(entity.name) } as AgentEvent
      return // Short-circuit the loop — delegation is the entire response
    }

    if (delegationIntent.type === 'revoke') {
      await revokeEntityMandate(supabase, orgId, entity.id, 'chat')
      // Cancel in-flight async tasks for this entity (Phase 39 integration)
      // TODO: Phase 39 provides cancelEntityTasks(supabase, orgId, entityId)

      return
    }
  } else {
    // Entity not found — let the normal loop handle it
    logger.warn('[taor-loop] Delegation intent detected but entity not resolved', {
      mention: delegationIntent.entityMention,
    })
  }
}
```
