  supabase: SupabaseClient,

    params.agentConfig,
    params.orgSettings,
    undefined,
    params.calibratedThresholds,
    params.delegationMandate,

    // For auto_delegated: log to delegation audit trail
    if (routing.decision === 'auto_delegated' && params.delegationMandate) {
      import('./delegation-mandate').then(({ logDelegatedAction }) => {
        logDelegatedAction(supabase, {
          orgId: params.org_id,
          entityId: params.delegationMandate!.entity_id,
          mandateId: params.delegationMandate!.id,
          actionType: params.action_type,
          actionSummary: params.action_summary,
          actionPayload: params.action_payload,
          agentRunId: params.agent_run_id ?? undefined,
        }).catch(err => {
          logger.warn('[approval-queue] Failed to log delegated action:', err)
        })
      }).catch(() => {})

    return null
  }

  return createApproval(supabase, {
    org_id: params.org_id,

    agent_run_id: params.agent_run_id,
    action_type: params.action_type,
    action_payload: params.action_payload,
    action_summary: params.action_summary,
    confidence_score: params.confidence_score,
