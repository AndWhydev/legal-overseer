```typescript
/** Get full delegation history for an entity (all mandates, active and historical) */
export async function getEntityDelegationHistory(
  supabase: SupabaseClient,
  orgId: string,
  entityId: string,
): Promise<DelegationMandate[]> {
  const { data, error } = await supabase
    .from('delegation_mandates')
    .select('*')
    .eq('org_id', orgId)
    .eq('entity_id', entityId)
    .order('created_at', { ascending: false })

  if (error) {
    logger.error('[delegation] Failed to fetch delegation history', { orgId, entityId, error: error.message })
    return []
  }

  return (data ?? []) as DelegationMandate[]
}

/** Get all actions taken under a specific delegation mandate */
export async function getActionsForMandate(
  supabase: SupabaseClient,
  mandateId: string,
): Promise<DelegationActionEntry[]> {
  const { data, error } = await supabase
    .from('delegation_action_log')
    .select('*')
    .eq('mandate_id', mandateId)
    .order('created_at', { ascending: false })

  if (error) {
    logger.error('[delegation] Failed to fetch mandate actions', { mandateId, error: error.message })
    return []
  }

  return (data ?? []) as DelegationActionEntry[]
}

/** Get delegation audit summary for an entity (for display/reporting) */
export async function getDelegationAuditSummary(
  supabase: SupabaseClient,
  orgId: string,
  entityId: string,
): Promise<{
  currentMandate: DelegationMandate | null
  totalMandates: number
  totalActions: number
  totalFinancialImpact: number
  lastActionAt: string | null
}> {
  const currentMandate = await getEntityMandate(supabase, orgId, entityId)
  const history = await getEntityDelegationHistory(supabase, orgId, entityId)

  // Get all actions for this entity
  const { data: actions, error } = await supabase
    .from('delegation_action_log')
    .select('financial_impact, created_at')
    .eq('org_id', orgId)
    .eq('entity_id', entityId)
    .order('created_at', { ascending: false })

  const totalActions = actions?.length ?? 0
  const totalFinancialImpact = (actions ?? [])
    .filter((a: { financial_impact: number | null }) => a.financial_impact != null)
    .reduce((sum: number, a: { financial_impact: number | null }) => sum + (a.financial_impact ?? 0), 0)
  const lastActionAt = actions?.[0]?.created_at ?? null

  return {
    currentMandate,
    totalMandates: history.length,
    totalActions,
    totalFinancialImpact,
    lastActionAt: lastActionAt as string | null,
  }
}
```
