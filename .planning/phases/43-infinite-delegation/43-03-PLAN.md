```typescript
async function fetchDelegatedActionItems(
  supabase: SupabaseClient,
  orgId: string,
): Promise<string[]> {
  try {
    const { getRecentDelegatedActions } = await import('../agent/delegation-mandate')
    const since = new Date(Date.now() - 24 * 60 * 60 * 1000)
    const actions = await getRecentDelegatedActions(supabase, orgId, since)

    if (actions.length === 0) return []

    // Group by entity
    const entityMap = new Map<string, typeof actions>()
    for (const action of actions) {
      const existing = entityMap.get(action.entity_id) ?? []
      existing.push(action)
      entityMap.set(action.entity_id, existing)
    }

    // Build conversational summaries per entity
    const items: string[] = []
    for (const [entityId, entityActions] of entityMap) {
      // Resolve entity name
      const { data: entity } = await supabase
        .from('entity_nodes')
        .select('name')
        .eq('id', entityId)
        .single()

      const name = entity?.name ?? 'Unknown entity'
      const actionSummaries = entityActions.map(a => a.action_summary).join(', ')
      const totalImpact = entityActions
        .filter(a => a.financial_impact != null)
        .reduce((sum, a) => sum + (a.financial_impact ?? 0), 0)

      let line = `${name}: ${actionSummaries}`
      if (totalImpact > 0) {
        line += ` ($${totalImpact.toLocaleString()} impact)`
      }
      items.push(line)
    }

    return items
  } catch {
    return []
  }
}
```

The conversational format is achieved by the entity-grouped summaries. Example output:
"Steve: sent the invoice ($2,400), followed up on the DNS issue, scheduled the review call ($2,400 impact)"
