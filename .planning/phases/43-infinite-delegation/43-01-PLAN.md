<objective>
Create the `delegation_mandates` and `delegation_action_log` database tables and a TypeScript service layer for managing entity delegation mandates. This is the foundation that all other delegation plans depend on.
</objective>

<must_haves>

- `delegation_action_log` table for audit trail of autonomous actions
- TypeScript service: getEntityMandate, setEntityMandate, revokeEntityMandate, logDelegatedAction
- Unique constraint preventing duplicate active mandates per entity
- RLS policies scoped to org_id
</must_haves>

```sql
-- delegation_mandates: per-entity delegation level
CREATE TABLE delegation_mandates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  entity_id UUID NOT NULL REFERENCES entity_nodes(id) ON DELETE CASCADE,
  mandate_level TEXT NOT NULL CHECK (mandate_level IN ('infinite_autopilot', 'supervised', 'standard')),
  activated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  activated_via TEXT NOT NULL CHECK (activated_via IN ('chat', 'dashboard', 'api')),
  deactivated_at TIMESTAMPTZ,
  deactivated_via TEXT CHECK (deactivated_via IN ('chat', 'dashboard', 'api') OR deactivated_via IS NULL),
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Only one active mandate per entity per org
CREATE UNIQUE INDEX idx_delegation_mandates_active
  ON delegation_mandates (org_id, entity_id)
  WHERE deactivated_at IS NULL;

-- Index for lookups by entity
CREATE INDEX idx_delegation_mandates_entity
  ON delegation_mandates (entity_id, org_id);

-- RLS
ALTER TABLE delegation_mandates ENABLE ROW LEVEL SECURITY;
CREATE POLICY "org_isolation" ON delegation_mandates
  USING (org_id = (current_setting('app.current_org_id', true))::uuid);

-- delegation_action_log: audit trail for autonomous actions
CREATE TABLE delegation_action_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  entity_id UUID NOT NULL REFERENCES entity_nodes(id) ON DELETE CASCADE,
  mandate_id UUID NOT NULL REFERENCES delegation_mandates(id) ON DELETE CASCADE,
  action_type TEXT NOT NULL,
  action_summary TEXT NOT NULL,
  action_payload JSONB DEFAULT '{}',
  financial_impact NUMERIC,
  evidence_urls TEXT[] DEFAULT '{}',
  fiduciary_evaluation JSONB,
  agent_run_id UUID,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Index for morning briefing queries (last 24h per entity)
CREATE INDEX idx_delegation_action_log_briefing
  ON delegation_action_log (org_id, entity_id, created_at DESC);

-- RLS
ALTER TABLE delegation_action_log ENABLE ROW LEVEL SECURITY;
CREATE POLICY "org_isolation" ON delegation_action_log
  USING (org_id = (current_setting('app.current_org_id', true))::uuid);
```

```typescript
import type { SupabaseClient } from '@supabase/supabase-js'
import { logger } from '@/lib/core/logger'

export type MandateLevel = 'infinite_autopilot' | 'supervised' | 'standard'
export type ActivationChannel = 'chat' | 'dashboard' | 'api'

export interface DelegationMandate {
  id: string
  org_id: string
  entity_id: string
  mandate_level: MandateLevel
  activated_at: string
  activated_via: ActivationChannel
  deactivated_at: string | null
  deactivated_via: ActivationChannel | null
  created_at: string
  updated_at: string
}

export interface DelegationActionEntry {
  id: string
  org_id: string
  entity_id: string
  mandate_id: string
  action_type: string
  action_summary: string
  action_payload: Record<string, unknown>
  financial_impact: number | null
  evidence_urls: string[]
  fiduciary_evaluation: Record<string, unknown> | null
  agent_run_id: string | null
  created_at: string
}

/** Get the active delegation mandate for an entity (null = standard/no delegation) */
export async function getEntityMandate(
  supabase: SupabaseClient,
  orgId: string,
  entityId: string,
): Promise<DelegationMandate | null> {
  const { data, error } = await supabase
    .from('delegation_mandates')
    .select('*')
    .eq('org_id', orgId)
    .eq('entity_id', entityId)
    .is('deactivated_at', null)
    .single()

  if (error || !data) return null
  return data as DelegationMandate
}

/** Set or update the delegation mandate for an entity */
export async function setEntityMandate(
  supabase: SupabaseClient,
  orgId: string,
  entityId: string,
  level: MandateLevel,
  via: ActivationChannel,
): Promise<DelegationMandate> {
  // Deactivate any existing active mandate first
  await revokeEntityMandate(supabase, orgId, entityId, via)

  if (level === 'standard') {
    // standard = no active mandate needed
    return {
      id: '',
      org_id: orgId,
      entity_id: entityId,
      mandate_level: 'standard',
      activated_at: new Date().toISOString(),
      activated_via: via,
      deactivated_at: null,
      deactivated_via: null,
      created_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),
    }
  }

  const { data, error } = await supabase
    .from('delegation_mandates')
    .insert({
      org_id: orgId,
      entity_id: entityId,
      mandate_level: level,
      activated_via: via,
    })
    .select('*')
    .single()

  if (error || !data) {
    throw new Error(error?.message ?? 'Failed to set delegation mandate')
  }

  logger.info('[delegation] Mandate set', { orgId, entityId, level, via })
  return data as DelegationMandate
}

/** Revoke active delegation mandate (returns entity to standard routing) */
export async function revokeEntityMandate(
  supabase: SupabaseClient,
  orgId: string,
  entityId: string,
  via: ActivationChannel,
): Promise<boolean> {
  const { data, error } = await supabase
    .from('delegation_mandates')
    .update({
      deactivated_at: new Date().toISOString(),
      deactivated_via: via,
      updated_at: new Date().toISOString(),
    })
    .eq('org_id', orgId)
    .eq('entity_id', entityId)
    .is('deactivated_at', null)
    .select('id')

  if (error) {
    logger.error('[delegation] Revocation failed', { orgId, entityId, error: error.message })
    return false
  }

  const revoked = (data?.length ?? 0) > 0
  if (revoked) {
    logger.info('[delegation] Mandate revoked', { orgId, entityId, via })
  }
  return revoked
}

/** Log an autonomous action taken under delegation */
export async function logDelegatedAction(
  supabase: SupabaseClient,
  params: {
    orgId: string
    entityId: string
    mandateId: string
    actionType: string
    actionSummary: string
    actionPayload?: Record<string, unknown>
    financialImpact?: number
    evidenceUrls?: string[]
    fiduciaryEvaluation?: Record<string, unknown>
    agentRunId?: string
  },
): Promise<DelegationActionEntry> {
  const { data, error } = await supabase
    .from('delegation_action_log')
    .insert({
      org_id: params.orgId,
      entity_id: params.entityId,
      mandate_id: params.mandateId,
      action_type: params.actionType,
      action_summary: params.actionSummary,
      action_payload: params.actionPayload ?? {},
      financial_impact: params.financialImpact ?? null,
      evidence_urls: params.evidenceUrls ?? [],
      fiduciary_evaluation: params.fiduciaryEvaluation ?? null,
      agent_run_id: params.agentRunId ?? null,
    })
    .select('*')
    .single()

  if (error || !data) {
    throw new Error(error?.message ?? 'Failed to log delegated action')
  }

  return data as DelegationActionEntry
}

/** Get recent autonomous actions for delegated entities (for morning briefing) */
export async function getRecentDelegatedActions(
  supabase: SupabaseClient,
  orgId: string,
  since: Date,
): Promise<DelegationActionEntry[]> {
  const { data, error } = await supabase
    .from('delegation_action_log')
    .select('*')
    .eq('org_id', orgId)
    .gte('created_at', since.toISOString())
    .order('created_at', { ascending: false })

  if (error) {
    logger.error('[delegation] Failed to fetch recent actions', { orgId, error: error.message })
    return []
  }

  return (data ?? []) as DelegationActionEntry[]
}

/** Check if an entity is delegated at infinite_autopilot level */
export async function isEntityFullyDelegated(
  supabase: SupabaseClient,
  orgId: string,
  entityId: string,
): Promise<boolean> {
  const mandate = await getEntityMandate(supabase, orgId, entityId)
  return mandate?.mandate_level === 'infinite_autopilot'
}
```
