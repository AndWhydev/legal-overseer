# Plan 20-01: Role Schema & Type System

## Goal
Define the database schema and TypeScript types for the role engine. This is the data foundation everything else builds on.

## Context
- Existing: `agent_configs` table, `AgentConfig` interface, `AgentType` union, `agent_runs` for logging
- Pattern: DI via SupabaseClient param, JSONB for flexible configs, RLS per org_id
- Extends: `bitbit-core/types.ts` for types, new migration for tables

## Tasks

### Task 1: Database Migration — Role Tables
Create migration `092_role_engine_tables.sql` with:

```sql
-- Role type enum
CREATE TYPE role_type AS ENUM ('finance', 'comms', 'sales');

-- Autonomy level enum
CREATE TYPE autonomy_level AS ENUM ('observer', 'copilot', 'autopilot');

-- Role configurations (one per role per org)
CREATE TABLE role_configs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id),
  role_type role_type NOT NULL,
  enabled BOOLEAN NOT NULL DEFAULT false,
  autonomy_level autonomy_level NOT NULL DEFAULT 'copilot',
  config JSONB NOT NULL DEFAULT '{}',  -- role-specific settings
  tick_interval_seconds INTEGER NOT NULL DEFAULT 300,  -- how often role ticks
  daily_budget_cents INTEGER NOT NULL DEFAULT 500,  -- $5/day default cap
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(org_id, role_type)
);

-- Role persistent state (working memory)
CREATE TABLE role_states (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role_config_id UUID NOT NULL REFERENCES role_configs(id) ON DELETE CASCADE,
  org_id UUID NOT NULL REFERENCES organizations(id),
  state JSONB NOT NULL DEFAULT '{}',  -- FSM state + working data
  version INTEGER NOT NULL DEFAULT 1,  -- optimistic concurrency
  last_tick_at TIMESTAMPTZ,
  next_tick_at TIMESTAMPTZ,
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE(role_config_id)
);

-- Multi-step workflow tracking
CREATE TABLE role_workflows (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role_config_id UUID NOT NULL REFERENCES role_configs(id) ON DELETE CASCADE,
  org_id UUID NOT NULL REFERENCES organizations(id),
  workflow_type TEXT NOT NULL,  -- e.g. 'invoice_reminder_sequence', 'lead_nurture'
  status TEXT NOT NULL DEFAULT 'pending' CHECK (status IN ('pending','active','paused','completed','failed','cancelled')),
  steps JSONB NOT NULL DEFAULT '[]',  -- [{step_id, name, status, result, started_at, completed_at}]
  current_step INTEGER NOT NULL DEFAULT 0,
  context JSONB NOT NULL DEFAULT '{}',  -- workflow-scoped data
  error TEXT,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  next_step_at TIMESTAMPTZ,  -- when to resume (for time-delayed steps)
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Role activity log (what the role did/found/learned)
CREATE TABLE role_activity (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role_config_id UUID NOT NULL REFERENCES role_configs(id) ON DELETE CASCADE,
  org_id UUID NOT NULL REFERENCES organizations(id),
  activity_type TEXT NOT NULL CHECK (activity_type IN ('insight','action','escalation','learning','error','workflow_step')),
  summary TEXT NOT NULL,
  details JSONB NOT NULL DEFAULT '{}',
  confidence REAL,
  autonomy_mode autonomy_level,
  reasoning TEXT,  -- why the role decided to do this
  reversible BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Business intelligence cache
CREATE TABLE bi_snapshots (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id),
  metric_type TEXT NOT NULL,  -- revenue_radar, client_health, cash_flow, capacity
  data JSONB NOT NULL DEFAULT '{}',
  computed_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  expires_at TIMESTAMPTZ NOT NULL DEFAULT (now() + INTERVAL '24 hours'),
  UNIQUE(org_id, metric_type)
);

-- RLS policies
ALTER TABLE role_configs ENABLE ROW LEVEL SECURITY;
ALTER TABLE role_states ENABLE ROW LEVEL SECURITY;
ALTER TABLE role_workflows ENABLE ROW LEVEL SECURITY;
ALTER TABLE role_activity ENABLE ROW LEVEL SECURITY;
ALTER TABLE bi_snapshots ENABLE ROW LEVEL SECURITY;

CREATE POLICY role_configs_org ON role_configs FOR ALL USING (org_id = (current_setting('app.current_org_id', true))::uuid);
CREATE POLICY role_states_org ON role_states FOR ALL USING (org_id = (current_setting('app.current_org_id', true))::uuid);
CREATE POLICY role_workflows_org ON role_workflows FOR ALL USING (org_id = (current_setting('app.current_org_id', true))::uuid);
CREATE POLICY role_activity_org ON role_activity FOR ALL USING (org_id = (current_setting('app.current_org_id', true))::uuid);
CREATE POLICY bi_snapshots_org ON bi_snapshots FOR ALL USING (org_id = (current_setting('app.current_org_id', true))::uuid);

-- Service role bypass for cron/workers
CREATE POLICY role_configs_service ON role_configs FOR ALL TO service_role USING (true);
CREATE POLICY role_states_service ON role_states FOR ALL TO service_role USING (true);
CREATE POLICY role_workflows_service ON role_workflows FOR ALL TO service_role USING (true);
CREATE POLICY role_activity_service ON role_activity FOR ALL TO service_role USING (true);
CREATE POLICY bi_snapshots_service ON bi_snapshots FOR ALL TO service_role USING (true);

-- Indexes
CREATE INDEX idx_role_configs_org ON role_configs(org_id);
CREATE INDEX idx_role_states_config ON role_states(role_config_id);
CREATE INDEX idx_role_workflows_config_status ON role_workflows(role_config_id, status);
CREATE INDEX idx_role_workflows_next_step ON role_workflows(next_step_at) WHERE status = 'active';
CREATE INDEX idx_role_activity_config_created ON role_activity(role_config_id, created_at DESC);
CREATE INDEX idx_role_activity_org_type ON role_activity(org_id, activity_type, created_at DESC);
CREATE INDEX idx_bi_snapshots_org_metric ON bi_snapshots(org_id, metric_type);
```

Also add `role_id` column to `semantic_memories`:
```sql
ALTER TABLE semantic_memories ADD COLUMN role_id UUID REFERENCES role_configs(id);
CREATE INDEX idx_semantic_memories_role ON semantic_memories(role_id) WHERE role_id IS NOT NULL;
```

### Task 2: TypeScript Types
Add to `personal-assistant/src/lib/bitbit-core/types.ts`:

```typescript
// --- Role System ---

export type RoleType = 'finance' | 'comms' | 'sales'
export type AutonomyLevel = 'observer' | 'copilot' | 'autopilot'

export interface RoleConfig {
  id: string
  org_id: string
  role_type: RoleType
  enabled: boolean
  autonomy_level: AutonomyLevel
  config: Record<string, unknown>
  tick_interval_seconds: number
  daily_budget_cents: number
  created_at: string
  updated_at: string
}

export interface RoleState {
  id: string
  role_config_id: string
  org_id: string
  state: Record<string, unknown>
  version: number
  last_tick_at: string | null
  next_tick_at: string | null
  updated_at: string
}

export type WorkflowStatus = 'pending' | 'active' | 'paused' | 'completed' | 'failed' | 'cancelled'

export interface WorkflowStep {
  step_id: string
  name: string
  status: 'pending' | 'active' | 'completed' | 'failed' | 'skipped'
  result?: unknown
  started_at?: string
  completed_at?: string
}

export interface RoleWorkflow {
  id: string
  role_config_id: string
  org_id: string
  workflow_type: string
  status: WorkflowStatus
  steps: WorkflowStep[]
  current_step: number
  context: Record<string, unknown>
  error?: string
  started_at: string | null
  completed_at: string | null
  next_step_at: string | null
  created_at: string
  updated_at: string
}

export type ActivityType = 'insight' | 'action' | 'escalation' | 'learning' | 'error' | 'workflow_step'

export interface RoleActivity {
  id: string
  role_config_id: string
  org_id: string
  activity_type: ActivityType
  summary: string
  details: Record<string, unknown>
  confidence?: number
  autonomy_mode?: AutonomyLevel
  reasoning?: string
  reversible?: boolean
  created_at: string
}

export interface BISnapshot {
  id: string
  org_id: string
  metric_type: string
  data: Record<string, unknown>
  computed_at: string
  expires_at: string
}
```

Update `bitbit-core/index.ts` to export new types.
