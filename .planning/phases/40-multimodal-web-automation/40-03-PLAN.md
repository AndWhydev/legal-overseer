# Plan 40-03: Browser Task Engine + Async Task Integration

<objective>
Create the browser task execution engine that orchestrates a complete browser automation session: pre-flight checks (domain gate + budget check + credential availability), Stagehand session lifecycle, credential injection, autonomous task execution, and result reporting. This engine runs browser jobs as async tasks within the Phase 39 infrastructure and enforces ephemeral session isolation (CUA-05) and configurable autonomy behavior (CUA-06).
</objective>

```typescript
/**
 * Browser task execution engine.
 * Phase 40: Multimodal Web Automation
 *
 * Orchestrates the full lifecycle of a browser automation task:
 * 1. Pre-flight checks (CUA-11): domain gate, budget check, credential availability
 * 2. Session creation: ephemeral Browserbase session via Stagehand (CUA-03, CUA-05)
 * 3. Navigation + credential injection (CUA-08)
 * 4. Autonomous task execution via Stagehand agent() (CUA-06)
 * 5. Cost monitoring throughout (CUA-10)
 * 6. Result collection + session close
 *
 * Sessions are ephemeral: no data persists between tasks or orgs.
 * Browserbase handles container isolation -- we don't manage containers.
 */

import { logger } from '@/lib/core/logger'
import { createSession, navigateTo, act, extract, closeSession } from './stagehand-client'
import { checkDomainAuthorization } from './domain-gate'
import { injectCredentials } from './credential-injector'
import { createCostBudget, recordTokens, recordSessionTime, checkBudget } from './cost-monitor'
import type { BrowserTaskParams, BrowserTaskResult, BrowserTaskStatus } from './types'
import type { SupabaseClient } from '@supabase/supabase-js'

// ---------------------------------------------------------------------------
// Constants
// ---------------------------------------------------------------------------

/** Default maximum session duration: 5 minutes. */
const DEFAULT_MAX_DURATION_SECONDS = 300

/** Estimated tokens per Stagehand action (act/observe/extract). */
const ESTIMATED_TOKENS_PER_ACTION = 2000

/** Cost monitoring interval: check budget every N actions. */
const COST_CHECK_INTERVAL = 3

// ---------------------------------------------------------------------------
// Pre-flight checks (CUA-11: fail-closed gate)
// ---------------------------------------------------------------------------

interface PreFlightResult {
  passed: boolean
  failReason?: string
}

async function runPreFlightChecks(
  params: BrowserTaskParams,
  supabase: SupabaseClient,
  ltvMultiplier: number,
): Promise<PreFlightResult> {
  // Check 1: Domain authorization
  const domainResult = await checkDomainAuthorization(params.url, params.orgId, supabase)
  if (!domainResult.allowed) {
    return { passed: false, failReason: `Domain check failed: ${domainResult.reason}` }
  }

  // Check 2: Budget authorization (verify LTV multiplier is valid)
  const budget = createCostBudget(ltvMultiplier)
  if (budget.maxCostUsd <= 0) {
    return { passed: false, failReason: 'Budget check failed: entity has zero or negative budget' }
  }

  // Check 3: Credential availability (if authenticated navigation requested)
  if (params.credentialSource === 'composio' && !params.composioConnectionId) {
    return { passed: false, failReason: 'Credential check failed: composioConnectionId required for Composio auth' }
  }
  if (params.credentialSource === '1password' && !params.opSecretRef) {
    return { passed: false, failReason: 'Credential check failed: opSecretRef required for 1Password auth' }
  }

  return { passed: true }
}

// ---------------------------------------------------------------------------
// Main execution function
// ---------------------------------------------------------------------------

/**
 * Execute a browser automation task end-to-end.
 *
 * This is the function called by spawn_browser_agent via the tool executor.
 * It manages the complete session lifecycle and returns structured results
 * for inline display in the conversation (D-10).
 */
export async function executeBrowserTask(
  params: BrowserTaskParams,
  options?: {
    supabase?: SupabaseClient
    ltvMultiplier?: number
  },
): Promise<BrowserTaskResult> {
  const supabase = options?.supabase
  const ltvMultiplier = options?.ltvMultiplier ?? 1.0
  const maxDuration = params.maxDurationSeconds ?? DEFAULT_MAX_DURATION_SECONDS

  logger.info('browser.task.starting', {
    url: params.url,
    objective: params.objective,
    orgId: params.orgId,
    credentialSource: params.credentialSource,
  })

  // ── Pre-flight checks (CUA-11) ──────────────────────────────────────
  if (supabase) {
    const preFlight = await runPreFlightChecks(params, supabase, ltvMultiplier)
    if (!preFlight.passed) {
      logger.warn('browser.task.preflight_failed', { reason: preFlight.failReason })
      return {
        status: 'failed',
        sessionId: '',
        replayUrl: '',
        summary: `Browser task blocked by pre-flight check: ${preFlight.failReason}`,
        actions: [],
        durationSeconds: 0,
        estimatedCost: { llmTokens: 0, sessionMinutes: 0 },
        error: preFlight.failReason,
      }
    }
  }

  // ── Create ephemeral session (CUA-03, CUA-05) ──────────────────────
  const session = await createSession({ recordSession: true })
  const costBudget = createCostBudget(ltvMultiplier)
  const startTime = Date.now()
  let status: BrowserTaskStatus = 'running'

  try {
    // ── Navigate to target URL ────────────────────────────────────────
    await navigateTo(session, params.url)
    recordTokens(costBudget, ESTIMATED_TOKENS_PER_ACTION)

    // ── Credential injection (CUA-08) ─────────────────────────────────
    if (params.credentialSource && params.credentialSource !== 'none') {
      const credResult = await injectCredentials(session, params.credentialSource, {
        composioConnectionId: params.composioConnectionId,
        opSecretRef: params.opSecretRef,
      })
      recordTokens(costBudget, ESTIMATED_TOKENS_PER_ACTION * 2) // login usually takes 2+ actions

      if (!credResult.success) {
        throw new Error(`Credential injection failed: ${credResult.error}`)
      }
    }

    // ── Autonomous task execution (CUA-06) ────────────────────────────
    // Use Stagehand agent() for multi-step autonomous navigation.
    // The agent acts fully autonomously without confirmation prompts,
    // consistent with Phase 38's "pure intelligence" pattern.
    const taskResult = await session.stagehand.agent({
      task: params.objective,
      maxSteps: 20,
    })

    // Record estimated token consumption for the agent execution
    const estimatedAgentTokens = ESTIMATED_TOKENS_PER_ACTION * (taskResult?.steps?.length ?? 10)
    recordTokens(costBudget, estimatedAgentTokens)

    // Check cost budget after agent execution
    const costCheck = checkBudget(costBudget)
    if (!costCheck.withinBudget) {
      status = 'budget_exceeded'
      logger.warn('browser.task.budget_exceeded', {
        sessionId: session.sessionId,
        currentCost: costCheck.currentCostUsd,
        maxCost: costCheck.maxCostUsd,
      })
    } else {
      status = 'completed'
    }

    // ── Extract results if applicable ─────────────────────────────────
    let extractedData: Record<string, unknown> | undefined
    try {
      extractedData = await extract(session, `Summarize the results of: ${params.objective}`)
      recordTokens(costBudget, ESTIMATED_TOKENS_PER_ACTION)
    } catch {
      // Extraction is optional -- don't fail the task if it doesn't work
      logger.info('browser.task.extract_skipped', { sessionId: session.sessionId })
    }

    // ── Close session and return results ──────────────────────────────
    const sessionResult = await closeSession(session)
    const durationSeconds = Math.round((Date.now() - startTime) / 1000)

    // Record session time cost
    recordSessionTime(costBudget, Math.ceil(durationSeconds / 60))

    const summary = typeof taskResult === 'string'
      ? taskResult
      : `Completed browser task: ${params.objective}`

    return {
      status,
      sessionId: sessionResult.sessionId,
      replayUrl: sessionResult.replayUrl,
      summary,
      extractedData,
      actions: session.actions,
      durationSeconds,
      estimatedCost: {
        llmTokens: costBudget.llmTokensConsumed,
        sessionMinutes: costBudget.sessionMinutes,
      },
    }
  } catch (error) {
    // ── Error handling: close session gracefully ──────────────────────
    const message = error instanceof Error ? error.message : String(error)
    logger.error('browser.task.failed', {
      sessionId: session.sessionId,
      error: message,
    })

    let sessionResult: { sessionId: string; replayUrl: string } = {
      sessionId: session.sessionId,
      replayUrl: `https://browserbase.com/sessions/${session.sessionId}`,
    }

    try {
      const closed = await closeSession(session)
      sessionResult = { sessionId: closed.sessionId, replayUrl: closed.replayUrl }
    } catch {
      // Session close failed -- use constructed URL
    }

    const durationSeconds = Math.round((Date.now() - startTime) / 1000)

    return {
      status: 'failed',
      sessionId: sessionResult.sessionId,
      replayUrl: sessionResult.replayUrl,
      summary: `Browser task failed: ${message}`,
      actions: session.actions,
      durationSeconds,
      estimatedCost: {
        llmTokens: costBudget.llmTokensConsumed,
        sessionMinutes: Math.ceil(durationSeconds / 60),
      },
      error: message,
    }
  }
}
```
