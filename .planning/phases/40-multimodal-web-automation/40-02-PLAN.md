```typescript
/**
 * Domain authorization gate for browser automation.
 * Phase 40: CUA-04 (per-org domain configuration), CUA-11 (fail-closed pre-flight).
 *
 * Default: OPEN access -- any domain is allowed.
 * Orgs can optionally BLOCK specific domains.
 * This is an inverse allowlist -- blocklist model.
 */

import { logger } from '@/lib/core/logger'
import type { SupabaseClient } from '@supabase/supabase-js'

export interface DomainGateResult {
  allowed: boolean
  reason?: string
  domain: string
}

/**
 * Extract the domain from a URL for authorization checking.
 */
export function extractDomain(url: string): string {
  try {
    return new URL(url).hostname.toLowerCase()
  } catch {
    throw new Error(`Invalid URL for domain extraction: ${url}`)
  }
}

/**
 * Check if the target domain is authorized for the given org.
 *
 * Policy: Open by default. Org-level blocklist stored in org_settings.blocked_domains (string[]).
 * If the domain matches any entry in the blocklist, navigation is denied.
 *
 * @returns DomainGateResult with allowed=true if navigation is permitted
 */
export async function checkDomainAuthorization(
  url: string,
  orgId: string,
  supabase: SupabaseClient,
): Promise<DomainGateResult> {
  const domain = extractDomain(url)

  // Fetch org-level domain blocklist
  const { data: orgSettings, error } = await supabase
    .from('organizations')
    .select('settings')
    .eq('id', orgId)
    .single()

  if (error) {
    logger.error('browser.domain_gate.org_fetch_failed', { orgId, error: error.message })
    // Fail closed: if we can't verify, deny access
    return { allowed: false, reason: `Failed to verify domain authorization: ${error.message}`, domain }
  }

  const blockedDomains: string[] = orgSettings?.settings?.blocked_domains ?? []

  // Check if domain or parent domain is blocked
  const isBlocked = blockedDomains.some(blocked => {
    const normalizedBlocked = blocked.toLowerCase().trim()
    return domain === normalizedBlocked || domain.endsWith(`.${normalizedBlocked}`)
  })

  if (isBlocked) {
    logger.warn('browser.domain_gate.blocked', { orgId, domain })
    return { allowed: false, reason: `Domain "${domain}" is blocked by organization policy`, domain }
  }

  logger.info('browser.domain_gate.allowed', { orgId, domain })
  return { allowed: true, domain }
}
```

```typescript
/**
 * Credential injection for browser automation.
 * Phase 40: CUA-08 (Composio + 1Password dual credential injection)
 *
 * SECURITY: Uses brokered credential pattern.
 * - Credentials are injected into the browser session via Stagehand act() calls
 * - The LLM prompt NEVER contains raw credentials
 * - Composio handles OAuth refresh and token management
 * - 1Password CLI retrieves secrets at runtime
 */

import { logger } from '@/lib/core/logger'
import type { StagehandSession } from './stagehand-client'

export interface CredentialSet {
  username?: string
  password?: string
  apiKey?: string
  oauthToken?: string
  custom?: Record<string, string>
}

export interface CredentialInjectionResult {
  success: boolean
  source: 'composio' | '1password' | 'none'
  error?: string
}

/**
 * Retrieve credentials from Composio managed authentication.
 * Composio handles OAuth flows, token refresh, and secure storage.
 */
async function getComposioCredentials(connectionId: string): Promise<CredentialSet> {
  // Dynamic import to avoid loading Composio when not needed
  const { getConnectionCredentials } = await import('@/lib/composio/mcp-session')
  const creds = await getConnectionCredentials(connectionId)
  return {
    username: creds.username,
    password: creds.password,
    oauthToken: creds.accessToken,
    apiKey: creds.apiKey,
  }
}

/**
 * Retrieve credentials from 1Password vault via CLI.
 * Secret reference format: "op://VaultName/ItemName/FieldName"
 */
async function get1PasswordCredentials(secretRef: string): Promise<CredentialSet> {
  const { execSync } = await import('child_process')

  try {
    // op read returns the secret value directly
    const value = execSync(`op read "${secretRef}"`, {
      encoding: 'utf-8',
      timeout: 10_000,
    }).trim()

    // Determine field type from reference path
    const field = secretRef.split('/').pop()?.toLowerCase() ?? ''

    if (field.includes('password') || field.includes('pass')) {
      return { password: value }
    } else if (field.includes('user') || field.includes('email')) {
      return { username: value }
    } else if (field.includes('key') || field.includes('api')) {
      return { apiKey: value }
    } else if (field.includes('token')) {
      return { oauthToken: value }
    }

    // Default: treat as password
    return { password: value }
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error)
    throw new Error(`1Password credential retrieval failed: ${message}`)
  }
}

/**
 * Inject credentials into the browser session.
 *
 * SECURITY: Credentials are passed directly to Stagehand act() calls,
 * NOT included in any LLM prompt. The model sees "fill in the username field"
 * and "fill in the password field" -- the actual values are injected at the
 * Stagehand level.
 */
export async function injectCredentials(
  session: StagehandSession,
  source: 'composio' | '1password' | 'none',
  options: {
    composioConnectionId?: string
    opSecretRef?: string
  },
): Promise<CredentialInjectionResult> {
  if (source === 'none') {
    return { success: true, source: 'none' }
  }

  try {
    let credentials: CredentialSet

    if (source === 'composio') {
      if (!options.composioConnectionId) {
        return { success: false, source, error: 'composioConnectionId required for Composio credential injection' }
      }
      credentials = await getComposioCredentials(options.composioConnectionId)
    } else if (source === '1password') {
      if (!options.opSecretRef) {
        return { success: false, source, error: 'opSecretRef required for 1Password credential injection' }
      }
      credentials = await get1PasswordCredentials(options.opSecretRef)
    } else {
      return { success: false, source, error: `Unknown credential source: ${source}` }
    }

    // Import act dynamically to avoid circular dependencies
    const { act } = await import('./stagehand-client')

    // Inject credentials into the page via Stagehand
    // The credential values go directly to the browser, not through the LLM
    if (credentials.username) {
      await act(session, `fill in the username or email field with "${credentials.username}"`)
    }
    if (credentials.password) {
      await act(session, `fill in the password field with "${credentials.password}"`)
    }

    logger.info('browser.credentials.injected', {
      source,
      sessionId: session.sessionId,
      hasUsername: !!credentials.username,
      hasPassword: !!credentials.password,
      hasApiKey: !!credentials.apiKey,
      hasOAuthToken: !!credentials.oauthToken,
    })

    return { success: true, source }
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error)
    logger.error('browser.credentials.failed', { source, error: message })
    return { success: false, source, error: message }
  }
}
```

```typescript
/**
 * Cost circuit breaker for browser automation sessions.
 * Phase 40: CUA-10 (adaptive limits based on entity LTV and task value)
 *
 * Tracks cumulative cost during a browser session and terminates
 * when the entity's LTV-scaled budget is exceeded.
 *
 * Budget formula: base_budget * ltv_multiplier
 * - base_budget: $0.50 per browser task (default)
 * - ltv_multiplier: from entity_overrides table (Phase 37, default 1.0)
 *
 * Cost components:
 * - LLM tokens: Stagehand AI calls (estimated from action count)
 * - Browserbase session time: billed per minute
 */

import { logger } from '@/lib/core/logger'

export interface CostBudget {
  /** Maximum allowed cost in USD for this browser task. */
  maxCostUsd: number
  /** Current accumulated cost in USD. */
  currentCostUsd: number
  /** LLM tokens consumed so far. */
  llmTokensConsumed: number
  /** Browserbase session minutes consumed. */
  sessionMinutes: number
}

export interface CostCheckResult {
  withinBudget: boolean
  currentCostUsd: number
  maxCostUsd: number
  remainingBudget: number
  utilizationPercent: number
}

/** Base budget per browser task in USD. */
const BASE_BUDGET_USD = 0.50

/** Estimated cost per 1000 LLM tokens in USD (Sonnet 4.6 blended rate). */
const COST_PER_1K_TOKENS = 0.009 // ~$3/MTok input + $15/MTok output blended

/** Cost per Browserbase session minute in USD (Developer tier). */
const COST_PER_SESSION_MINUTE = 0.002 // $0.12/hr = $0.002/min

/**
 * Create a new cost budget for a browser task.
 *
 * @param ltvMultiplier Entity LTV multiplier from entity_overrides (default 1.0)
 * @param baseBudgetOverride Optional base budget override in USD
 */
export function createCostBudget(
  ltvMultiplier: number = 1.0,
  baseBudgetOverride?: number,
): CostBudget {
  const baseBudget = baseBudgetOverride ?? BASE_BUDGET_USD
  return {
    maxCostUsd: baseBudget * Math.max(0.1, Math.min(10.0, ltvMultiplier)),
    currentCostUsd: 0,
    llmTokensConsumed: 0,
    sessionMinutes: 0,
  }
}

/**
 * Record LLM token consumption and check budget.
 */
export function recordTokens(budget: CostBudget, tokenCount: number): CostCheckResult {
  budget.llmTokensConsumed += tokenCount
  budget.currentCostUsd += (tokenCount / 1000) * COST_PER_1K_TOKENS
  return checkBudget(budget)
}

/**
 * Record session time and check budget.
 */
export function recordSessionTime(budget: CostBudget, minutes: number): CostCheckResult {
  budget.sessionMinutes += minutes
  budget.currentCostUsd += minutes * COST_PER_SESSION_MINUTE
  return checkBudget(budget)
}

/**
 * Check if the current cost is within budget.
 */
export function checkBudget(budget: CostBudget): CostCheckResult {
  const withinBudget = budget.currentCostUsd < budget.maxCostUsd
  const result: CostCheckResult = {
    withinBudget,
    currentCostUsd: Math.round(budget.currentCostUsd * 10000) / 10000,
    maxCostUsd: budget.maxCostUsd,
    remainingBudget: Math.round((budget.maxCostUsd - budget.currentCostUsd) * 10000) / 10000,
    utilizationPercent: Math.round((budget.currentCostUsd / budget.maxCostUsd) * 100),
  }

  if (!withinBudget) {
    logger.warn('browser.cost.budget_exceeded', {
      currentCostUsd: result.currentCostUsd,
      maxCostUsd: result.maxCostUsd,
      llmTokensConsumed: budget.llmTokensConsumed,
      sessionMinutes: budget.sessionMinutes,
    })
  } else if (result.utilizationPercent >= 80) {
    logger.info('browser.cost.high_utilization', {
      utilizationPercent: result.utilizationPercent,
      remainingBudget: result.remainingBudget,
    })
  }

  return result
}

/**
 * Pre-flight budget check: verify the entity has sufficient budget
 * before starting a browser session. Part of CUA-11 fail-closed gate.
 *
 * @param orgId Organization ID
 * @param entityId Entity ID for LTV lookup
 * @param ltvMultiplier Entity LTV multiplier (from entity_overrides, default 1.0)
 */
export function preFlightBudgetCheck(
  ltvMultiplier: number = 1.0,
): CostCheckResult {
  const budget = createCostBudget(ltvMultiplier)
  // Pre-flight always passes -- we're just validating the budget can be created
  // and the entity has a valid LTV multiplier
  return {
    withinBudget: true,
    currentCostUsd: 0,
    maxCostUsd: budget.maxCostUsd,
    remainingBudget: budget.maxCostUsd,
    utilizationPercent: 0,
  }
}
```
