```typescript
import type { ExecutionTask } from './types'

/**
 * Format a task's current state as a chat-friendly progress message.
 * This is the string that appears inline in the chat thread.
 */
export function formatTaskProgress(task: ExecutionTask): string {
  switch (task.status) {
    case 'pending':
      return `Queued: ${task.task_name}...`
    case 'claimed':
      return `Starting: ${task.task_name}...`
    case 'working': {
      const stepInfo = task.total_steps
        ? ` Step ${task.current_step}/${task.total_steps}`
        : ''
      const pct = task.progress_pct > 0 ? ` (${task.progress_pct}%)` : ''
      const msg = task.progress_message || task.task_name
      return `Working on ${msg}...${stepInfo}${pct}`
    }
    case 'paused':
      return `Paused: ${task.task_name} (${task.progress_pct}% complete)`
    case 'completed':
      return `Done: ${task.task_name}`
    case 'failed': {
      const retryInfo = task.retry_count < task.max_retries
        ? ` Retrying (attempt ${task.retry_count + 1}/${task.max_retries})...`
        : ' No more retries.'
      return `Failed: ${task.task_name} — ${task.error_message || 'Unknown error'}.${retryInfo}`
    }
    case 'cancelled': {
      if (task.partial_result) {
        return `Stopped: ${task.task_name}. Partial work preserved — ${task.progress_message || 'some steps completed'}.`
      }
      return `Cancelled: ${task.task_name}`
    }
    default:
      return task.task_name
  }
}

/**
 * Determine if a task update should generate a new chat message
 * vs updating an existing one.
 */
export function shouldCreateNewMessage(
  prevStatus: string | undefined,
  newStatus: string,
): boolean {
  // Terminal states always create a new final message
  if (newStatus === 'completed' || newStatus === 'cancelled') return true
  // First failure message
  if (newStatus === 'failed' && prevStatus !== 'failed') return true
  // Status change from non-working to working
  if (newStatus === 'working' && prevStatus !== 'working') return true
  // All other progress updates just update the existing message
  return false
}
```

```typescript
'use client'

import { useState, useCallback } from 'react'
import { useRealtimeSubscription } from '@/lib/realtime'
import type { RealtimePayload } from '@/lib/realtime'
import type { ExecutionTask } from '@/lib/agent/tasks/types'
import { formatTaskProgress } from '@/lib/agent/tasks/chat-progress'

export interface TaskProgressState {
  taskId: string
  status: string
  progressPct: number
  message: string
  isTerminal: boolean
}

/**
 * Subscribe to execution_tasks changes for a given thread.
 * Returns live progress state for all active tasks in the thread.
 */
export function useTaskProgress(threadId: string | null): TaskProgressState[] {
  const [tasks, setTasks] = useState<Map<string, TaskProgressState>>(new Map())

  const handleChange = useCallback((payload: RealtimePayload<ExecutionTask>) => {
    const task = payload.new
    if (!task?.id) return

    const isTerminal = task.status === 'completed' || task.status === 'cancelled' ||
      (task.status === 'failed' && task.retry_count >= task.max_retries)

    setTasks(prev => {
      const next = new Map(prev)
      next.set(task.id, {
        taskId: task.id,
        status: task.status,
        progressPct: task.progress_pct,
        message: formatTaskProgress(task),
        isTerminal,
      })
      // Clean up terminal tasks after 10 seconds
      if (isTerminal) {
        setTimeout(() => {
          setTasks(current => {
            const cleaned = new Map(current)
            cleaned.delete(task.id)
            return cleaned
          })
        }, 10000)
      }
      return next
    })
  }, [])

  useRealtimeSubscription<ExecutionTask>(
    'execution_tasks',
    {
      event: '*',
      filter: threadId ? `thread_id=eq.${threadId}` : undefined,
    },
    handleChange,
  )

  return Array.from(tasks.values())
}
```

```typescript
import type { SupabaseClient } from '@supabase/supabase-js'
import { cancelTask, getTasksByThread } from '@/lib/agent/tasks'
import { logger } from '@/lib/core/logger'

export const cancelTaskDefinition = {
  name: 'cancel_task',
  description: 'Cancel a running async task. If task_id is not provided, cancels the most recent active task in the current thread. Partial work is preserved.',
  input_schema: {
    type: 'object' as const,
    properties: {
      task_id: {
        type: 'string',
        description: 'Specific task ID to cancel. If omitted, cancels the most recent active task in the thread.',
      },
      reason: {
        type: 'string',
        description: 'Why the task is being cancelled (for audit trail).',
      },
    },
    required: [],
  },
}

export async function handleCancelTask(
  input: { task_id?: string; reason?: string },
  orgId: string,
  supabase: SupabaseClient,
  execOptions?: { threadId?: string },
): Promise<{ success: boolean; data?: unknown; error?: string }> {
  try {
    let taskId = input.task_id

    // If no task_id, find the most recent active task in the thread
    if (!taskId && execOptions?.threadId) {
      const activeTasks = await getTasksByThread(supabase, orgId, execOptions.threadId)
      const runningTasks = activeTasks.filter(t =>
        t.status === 'working' || t.status === 'claimed' || t.status === 'paused'
      )

      if (runningTasks.length === 0) {
        return { success: false, error: 'No active tasks found in this conversation.' }
      }

      if (runningTasks.length === 1) {
        taskId = runningTasks[0].id
      } else {
        // Multiple active tasks — return list for user disambiguation
        return {
          success: false,
          error: `Multiple active tasks found. Please specify which one:\n${
            runningTasks.map(t => `- ${t.task_name} (${t.status}, ${t.progress_pct}%)`).join('\n')
          }`,
          data: { activeTasks: runningTasks.map(t => ({ id: t.id, name: t.task_name, status: t.status })) },
        }
      }
    }

    if (!taskId) {
      return { success: false, error: 'No task_id provided and no thread context available.' }
    }

    const cancelled = await cancelTask(supabase, taskId, 'user')

    if (!cancelled) {
      return { success: false, error: 'Task not found or already in a terminal state.' }
    }

    const partialInfo = cancelled.partial_result
      ? ` Partial work preserved: ${cancelled.progress_message || 'some steps completed'}.`
      : ''

    logger.info('[cancel_task] Task cancelled', {
      taskId,
      taskName: cancelled.task_name,
      reason: input.reason,
    })

    return {
      success: true,
      data: {
        taskId: cancelled.id,
        taskName: cancelled.task_name,
        previousStatus: cancelled.status,
        progressAtCancellation: cancelled.progress_pct,
        partialResult: cancelled.partial_result,
        message: `Cancelled "${cancelled.task_name}".${partialInfo}`,
      },
    }
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err)
    logger.error('[cancel_task] Failed', { error: msg })
    return { success: false, error: msg }
  }
}
```
