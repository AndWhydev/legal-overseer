```typescript
/**
 * L1 Prompt Cache — Brain State Prefix Builder.
 * 
 * Structures the system prompt for Anthropic prompt caching:
 *   [CACHED PREFIX — cache_control: ephemeral]
 *     System identity + guidelines
 *     User profile + preferences
 *     Top entity dossiers (sorted by recency)
 *     Fiduciary constraints (always included)
 *     Domain profiles
 *   [DYNAMIC SUFFIX — changes per turn]
 *     Conversation-specific context
 *     WAL tail (unconsolidated signals)
 *     Action state
 * 
 * Cache TTL: ~1 hour (Anthropic's ephemeral cache).
 * Rebuild: hourly by Chief Librarian or on significant dossier change.
 * Target: ~50K tokens cached prefix -> $0.005/query read cost.
 */
import type { SupabaseClient } from '@supabase/supabase-js'
import { logger } from '@/lib/core/logger'
import type { EntityDossier, DomainProfile } from './types'

export const CACHE_PREFIX_BUDGET = 50_000  // tokens
const CHARS_PER_TOKEN = 3.5
const FIDUCIARY_RESERVED_TOKENS = 2000
const DOMAIN_PROFILE_MAX_TOKENS = 8000

export interface BrainStatePrefix {
  markdown: string
  tokenEstimate: number
  dossierCount: number
  constraintCount: number
  profileCount: number
  builtAt: string
}

export interface CacheableContext {
  cachedPrefix: Array<{ type: 'text'; text: string; cache_control: { type: 'ephemeral' } }>
  dynamicSuffix: Array<{ type: 'text'; text: string }>
}

function estimateTokens(text: string): number {
  return Math.ceil((text?.length ?? 0) / CHARS_PER_TOKEN)
}

export async function buildBrainStatePrefix(
  supabase: SupabaseClient,
  orgId: string,
  systemPrompt: string,
  opts?: { userProfile?: string; maxDossiers?: number }
): Promise<BrainStatePrefix> {
  const maxDossiers = opts?.maxDossiers ?? 15
  let remaining = CACHE_PREFIX_BUDGET

  // Section 1: System prompt (always included)
  const sections: string[] = [systemPrompt]
  remaining -= estimateTokens(systemPrompt)

  // Section 2: User profile (if provided)
  if (opts?.userProfile) {
    sections.push(`\n\n## User Profile\n${opts.userProfile}`)
    remaining -= estimateTokens(opts.userProfile) + 20
  }

  // Section 3: Fiduciary constraints (always included, reserved budget)
  const { data: constraints } = await supabase
    .from('memory_palace_entries')
    .select('content, entity_names')
    .eq('org_id', orgId)
    .eq('category', 'fiduciary_constraint')
    .eq('is_active', true)
    .order('created_at', { ascending: false })

  if (constraints?.length) {
    const constraintBlock = constraints
      .map(c => `- ${c.entity_names?.join(', ') ?? 'General'}: ${c.content}`)
      .join('\n')
    sections.push(`\n\n## Fiduciary Constraints\n${constraintBlock}`)
    remaining -= estimateTokens(constraintBlock) + 30
  }

  // Section 4: Top entity dossiers (sorted by last_compiled_at DESC)
  const { data: dossiers } = await supabase
    .from('entity_dossiers')
    .select('entity_name, dossier_markdown, token_count, last_compiled_at')
    .eq('org_id', orgId)
    .order('last_compiled_at', { ascending: false })
    .limit(maxDossiers)

  let dossierCount = 0
  if (dossiers?.length) {
    const dossierSections: string[] = []
    for (const d of dossiers) {
      const tokens = d.token_count || estimateTokens(d.dossier_markdown)
      if (remaining - tokens < DOMAIN_PROFILE_MAX_TOKENS) break  // reserve space for profiles
      dossierSections.push(`### ${d.entity_name}\n${d.dossier_markdown}`)
      remaining -= tokens + 10
      dossierCount++
    }
    if (dossierSections.length) {
      sections.push(`\n\n## Entity Dossiers\n${dossierSections.join('\n\n')}`)
    }
  }

  // Section 5: Domain profiles
  const { data: profiles } = await supabase
    .from('domain_profiles')
    .select('domain, profile_markdown, token_count')
    .eq('org_id', orgId)
    .order('last_compiled_at', { ascending: false })

  let profileCount = 0
  if (profiles?.length) {
    const profileSections: string[] = []
    for (const p of profiles) {
      const tokens = p.token_count || estimateTokens(p.profile_markdown)
      if (remaining - tokens < 0) break
      profileSections.push(`### ${p.domain}\n${p.profile_markdown}`)
      remaining -= tokens + 10
      profileCount++
    }
    if (profileSections.length) {
      sections.push(`\n\n## Domain Profiles\n${profileSections.join('\n\n')}`)
    }
  }

  const markdown = sections.join('')
  return {
    markdown,
    tokenEstimate: estimateTokens(markdown),
    dossierCount,
    constraintCount: constraints?.length ?? 0,
    profileCount,
    builtAt: new Date().toISOString(),
  }
}

export function splitCacheableContext(
  brainPrefix: BrainStatePrefix,
  dynamicContent: string
): CacheableContext {
  return {
    cachedPrefix: [{
      type: 'text' as const,
      text: brainPrefix.markdown,
      cache_control: { type: 'ephemeral' as const },
    }],
    dynamicSuffix: [{
      type: 'text' as const,
      text: dynamicContent,
    }],
  }
}
```
