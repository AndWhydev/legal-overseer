Create a single SQL migration with three tables:

**knowledge_log** (append-only WAL):
```sql
CREATE TABLE knowledge_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  entity_ids UUID[] DEFAULT '{}',
  signal_type TEXT NOT NULL CHECK (signal_type IN (
    'message', 'invoice', 'calendar', 'pattern', 'correction',
    'decision', 'relationship', 'pricing', 'fiduciary'
  )),
  content TEXT NOT NULL,
  confidence FLOAT DEFAULT 0.7 CHECK (confidence >= 0 AND confidence <= 1),
  source_memory_id UUID,
  source_thread_id UUID,
  consolidated_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);
```
Indexes:
- `idx_klog_unconsolidated ON knowledge_log (org_id, created_at DESC) WHERE consolidated_at IS NULL` (tail read query)
- `idx_klog_org_created ON knowledge_log (org_id, created_at DESC)` (general lookups)
- `idx_klog_entity_ids ON knowledge_log USING GIN (entity_ids)` (entity-scoped queries)

**entity_dossiers** (pre-compiled brain state):
```sql
CREATE TABLE entity_dossiers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  entity_id UUID NOT NULL REFERENCES entity_nodes(id) ON DELETE CASCADE,
  entity_name TEXT NOT NULL,
  dossier_markdown TEXT NOT NULL DEFAULT '',
  schema_json JSONB NOT NULL DEFAULT '{}',
  version INT NOT NULL DEFAULT 1,
  last_compiled_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  stale_since TIMESTAMPTZ,
  token_count INT NOT NULL DEFAULT 0,
  facts_incorporated INT NOT NULL DEFAULT 0,
  last_fact_id UUID,
  compilation_model TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE (org_id, entity_id)
);
```
Indexes:
- `idx_dossiers_stale ON entity_dossiers (org_id, stale_since) WHERE stale_since IS NOT NULL`
- `idx_dossiers_org_entity ON entity_dossiers (org_id, entity_id)`

**domain_profiles** (cross-entity rollups):
```sql
CREATE TABLE domain_profiles (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  domain TEXT NOT NULL CHECK (domain IN ('financial', 'relational', 'operational', 'behavioral')),
  profile_markdown TEXT NOT NULL DEFAULT '',
  constituent_hashes JSONB NOT NULL DEFAULT '{}',
  version INT NOT NULL DEFAULT 1,
  last_compiled_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  token_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE (org_id, domain)
);
```

export type SignalType =

  | 'invoice'
  | 'calendar'
  | 'pattern'
  | 'correction'
  | 'decision'
  | 'relationship'
  | 'pricing'
  | 'fiduciary'

export interface KnowledgeLogEntry {
  id: string
  org_id: string
  entity_ids: string[]
  signal_type: SignalType
  content: string
  confidence: number
  source_memory_id: string | null
  source_thread_id: string | null
  consolidated_at: string | null
  created_at: string
}

export interface EntityDossier {
  id: string
  org_id: string
  entity_id: string
  entity_name: string
  dossier_markdown: string
  schema_json: Record<string, unknown>
  version: number
  last_compiled_at: string
  stale_since: string | null
  token_count: number
  facts_incorporated: number
  last_fact_id: string | null
  compilation_model: string | null
  created_at: string
  updated_at: string
}

export interface DomainProfile {
  id: string
  org_id: string
  domain: DomainType
  profile_markdown: string
  constituent_hashes: Record<string, string>
  version: number
  last_compiled_at: string
  token_count: number
  created_at: string
  updated_at: string
}

// Worker queue types
export type WorkerTier = 'intake' | 'librarian' | 'chief'

export interface WALTailQuery {

export interface DossierDelta {

}
