```typescript
/**
 * Dossier Compiler — Delta-merge strategy for entity dossier updates.
 * 
 * CRITICAL: Uses delta-merge, NOT full re-summarization.
 * Naive re-summarization destroys ~60% of facts during compaction.
 * Instead: Current dossier + New facts -> LLM produces ONLY delta -> Merge
 * 
 * Model: Sonnet (~$0.003 per update)
 * Target: <10s per compilation
 * Max output: ~1500 tokens per dossier
 */
import { generateText } from 'ai'
import { models } from '@/lib/ai'
import { logger } from '@/lib/core/logger'
import type { EntityDossier, KnowledgeLogEntry, DossierDelta } from './types'

const MAX_DOSSIER_TOKENS = 1500
const CHARS_PER_TOKEN = 3.5

export function estimateTokenCount(text: string): number {
  if (!text) return 0
  return Math.ceil(text.length / CHARS_PER_TOKEN)
}

export async function compileDossierDelta(
  delta: DossierDelta
): Promise<{ markdown: string; token_count: number; model: string }> {
  const { current_dossier, new_facts } = delta
  const factsText = new_facts.map(f => `- [${f.signal_type}] ${f.content}`).join('\n')
  const model = models.balanced // Sonnet

  if (!current_dossier || !current_dossier.dossier_markdown) {
    // New dossier — create from scratch
    const { text } = await generateText({
      model,
      system: `You are a personal assistant's memory compiler. Create a structured dossier for an entity from the provided facts. Use this format:

## Summary
One paragraph overview of who/what this entity is and their relationship to the user.

## Key Facts
Bullet points of important permanent facts (role, company, contact info, preferences).

## Recent Activity
Bullet points of recent interactions and events, most recent first.

## Patterns
Any behavioral patterns, preferences, or recurring themes.

Keep the dossier under ${MAX_DOSSIER_TOKENS} tokens. Be concise but preserve all important details.`,
      prompt: `Entity: ${delta.entity_id}\n\nFacts:\n${factsText}`,
      maxTokens: 2000,
    })

    const tokenCount = estimateTokenCount(text)
    return { markdown: text, token_count: tokenCount, model: 'sonnet' }
  }

  // Delta merge — produce ONLY changes, then merge
  const { text } = await generateText({
    model,
    system: `You are updating an existing entity dossier with new facts. 
IMPORTANT: Output ONLY the updated dossier. Do not repeat unchanged information unless it provides context.
Merge new facts into the appropriate sections. If a new fact contradicts an existing one, update the existing fact and note the change.
Keep the dossier under ${MAX_DOSSIER_TOKENS} tokens. Preserve the section structure: ## Summary, ## Key Facts, ## Recent Activity, ## Patterns.`,
    prompt: `CURRENT DOSSIER:\n${current_dossier.dossier_markdown}\n\nNEW FACTS:\n${factsText}\n\nOutput the complete updated dossier:`,
    maxTokens: 2000,
  })

  const tokenCount = estimateTokenCount(text)
  return { markdown: text, token_count: tokenCount, model: 'sonnet' }
}
```

   a. Resolve entity_id from entity_name via `resolveEntityByAlias(supabase, job.org_id, job.entity_name)`. If not found, create a new entity_node.
   b. Load current dossier: `supabase.from('entity_dossiers').select('*').eq('org_id', job.org_id).eq('entity_id', entityId).single()`
   c. Load the actual WAL entries by fact_ids: `supabase.from('knowledge_log').select('*').in('id', job.fact_ids)`
   d. Call `compileDossierDelta({ entity_id: entityId, new_facts: walEntries, current_dossier: currentDossier })`
   e. Upsert into entity_dossiers:
      ```typescript
      await supabase.from('entity_dossiers').upsert({
        org_id: job.org_id,
        entity_id: entityId,
        entity_name: job.entity_name,
        dossier_markdown: result.markdown,
        schema_json: currentDossier?.schema_json ?? {},
        version: (currentDossier?.version ?? 0) + 1,
        last_compiled_at: new Date().toISOString(),
        stale_since: null,  // clear stale flag
        token_count: result.token_count,
        facts_incorporated: (currentDossier?.facts_incorporated ?? 0) + walEntries.length,
        last_fact_id: walEntries[walEntries.length - 1]?.id ?? currentDossier?.last_fact_id,
        compilation_model: result.model,
      }, { onConflict: 'org_id,entity_id' })
      ```
   f. After dossier update, fire Hebbian strengthening between this entity and all entities mentioned in the same WAL entries: call `hebbianStrengthen(supabase, job.org_id, entityId, coEntityId)` for each co-occurring entity.
