/**
 * Global Workspace — Competitive context selection.
 * Memory modules compete for context budget based on query relevance.
 * Implements Baars' Global Workspace Theory for attention allocation.
 */
import { logger } from '@/lib/core/logger'

export interface MemoryModule {
  name: string
  priority: 'critical' | 'high' | 'medium' | 'low'
  minTokens: number   // minimum allocation if selected
  maxTokens: number   // maximum allocation
  scoreRelevance: (query: string, context: ModuleContext) => number  // 0-1
}

export interface ModuleContext {
  entityMentions: string[]
  queryComplexity: 'system1' | 'system2'
  hasFinancialSignals: boolean
  hasDecisionSignals: boolean
  hasTemporalSignals: boolean
}

export interface ModuleAllocation {
  moduleName: string
  tokenBudget: number
  relevance: number
}

const PRIORITY_WEIGHTS: Record<string, number> = {
  critical: 10,
  high: 5,
  medium: 3,
  low: 1,
}

// Financial signal patterns
const FINANCIAL_PATTERNS = /\b(invoice|price|cost|budget|payment|bill|charge|fee|rate|quote)\b/i
// Decision signal patterns
const DECISION_PATTERNS = /\b(decide|decision|should|choose|option|alternative|trade.?off|pros?\s+and\s+cons?)\b/i
// Temporal signal patterns
const TEMPORAL_PATTERNS = /\b(when|before|after|last|recent|history|timeline|yesterday|today|tomorrow)\b/i

export const MEMORY_MODULES: MemoryModule[] = [
  {
    name: 'fiduciary',
    priority: 'critical',
    minTokens: 200,
    maxTokens: 500,
    scoreRelevance: () => 1.0,  // Always relevant (safety override)
  },
  {
    name: 'entity_dossier',
    priority: 'high',
    minTokens: 500,
    maxTokens: 4000,
    scoreRelevance: (_query, ctx) => {
      if (ctx.entityMentions.length === 0) return 0.2
      return Math.min(1.0, 0.5 + ctx.entityMentions.length * 0.15)
    },
  },
  {
    name: 'decision_memory',
    priority: 'high',
    minTokens: 200,
    maxTokens: 2000,
    scoreRelevance: (query, ctx) => {
      if (ctx.hasDecisionSignals || DECISION_PATTERNS.test(query)) return 0.8
      return 0.1
    },
  },
  {
    name: 'financial_memory',
    priority: 'medium',
    minTokens: 200,
    maxTokens: 2000,
    scoreRelevance: (query, ctx) => {
      if (ctx.hasFinancialSignals || FINANCIAL_PATTERNS.test(query)) return 0.8
      return 0.05
    },
  },
  {
    name: 'temporal_memory',
    priority: 'medium',
    minTokens: 500,
    maxTokens: 3000,
    scoreRelevance: (query, ctx) => {
      if (ctx.hasTemporalSignals || TEMPORAL_PATTERNS.test(query)) return 0.7
      return 0.3  // Always somewhat relevant (recent conversation)
    },
  },
  {
    name: 'pattern_memory',
    priority: 'medium',
    minTokens: 200,
    maxTokens: 1500,
    scoreRelevance: (_query, ctx) => {
      return ctx.entityMentions.length > 0 ? 0.4 : 0.1
    },
  },
  {
    name: 'warning_memory',
    priority: 'low',
    minTokens: 100,
    maxTokens: 500,
    scoreRelevance: () => 0.2,  // Low baseline, escalates via content
  },
]

export function allocateContextBudget(
  query: string,
  totalBudget: number,
  context: ModuleContext,
  modules: MemoryModule[] = MEMORY_MODULES
): ModuleAllocation[] {
  // Score all modules
  const scored = modules.map(m => ({
    module: m,
    relevance: m.scoreRelevance(query, context),
    effectiveScore: m.scoreRelevance(query, context) * PRIORITY_WEIGHTS[m.priority],
  }))

  // Filter out irrelevant modules (relevance < 0.1), but keep critical
  const eligible = scored.filter(s => s.relevance >= 0.1 || s.module.priority === 'critical')

  // Sort by effective score descending
  eligible.sort((a, b) => b.effectiveScore - a.effectiveScore)

  // Greedy allocation
  const allocations: ModuleAllocation[] = []
  let remaining = totalBudget

  for (const entry of eligible) {
    if (remaining <= 0) break
    const allocation = Math.min(entry.module.maxTokens, Math.max(entry.module.minTokens, Math.floor(remaining * (entry.effectiveScore / eligible.reduce((sum, e) => sum + e.effectiveScore, 0)))))
    const finalAlloc = Math.min(allocation, remaining)
    if (finalAlloc >= entry.module.minTokens) {
      allocations.push({
        moduleName: entry.module.name,
        tokenBudget: finalAlloc,
        relevance: entry.relevance,
      })
      remaining -= finalAlloc
    }
  }

  logger.debug('global-workspace', 'Budget allocated', {
    total: totalBudget,
    allocated: totalBudget - remaining,
    modules: allocations.map(a => `${a.moduleName}:${a.tokenBudget}`),
  })

  return allocations
}
