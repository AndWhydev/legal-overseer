/**
 * Brain Worker Infrastructure — BullMQ queue definitions and Redis connection.
 * Workers run on Fly.io alongside existing bridge machines.
 * Redis URL from REDIS_URL env var (Upstash or Fly Redis).
 */
import { Queue, Worker, type ConnectionOptions, type QueueOptions } from 'bullmq'
import { logger } from '@/lib/core/logger'

// Queue names for the 3-tier worker hierarchy
export const QUEUE_NAMES = {
  intake: 'memory:intake',
  financial: 'memory:financial',
  relational: 'memory:relational',
  operational: 'memory:operational',
  behavioral: 'memory:behavioral',
  synthesis: 'memory:synthesis',
} as const

export type QueueName = typeof QUEUE_NAMES[keyof typeof QUEUE_NAMES]

// Job types for each queue
export interface IntakeJob {
  org_id: string
  wal_entry_ids: string[]
  batch_window_start: string  // ISO date
}

export interface LibrarianJob {
  org_id: string
  entity_id: string
  fact_ids: string[]
  domain: 'financial' | 'relational' | 'operational' | 'behavioral'
}

export interface SynthesisJob {
  org_id: string
  trigger: 'batch_threshold' | 'scheduled' | 'manual'
  updated_entity_ids: string[]
}

export function getRedisConnection(): ConnectionOptions {
  const url = process.env.REDIS_URL
  if (!url) {
    logger.warn('brain-worker', 'REDIS_URL not set, using localhost:6379')
    return { host: 'localhost', port: 6379 }
  }
  // Parse redis:// or rediss:// URL
  const parsed = new URL(url)
  return {
    host: parsed.hostname,
    port: parseInt(parsed.port || '6379'),
    password: parsed.password || undefined,
    tls: parsed.protocol === 'rediss:' ? {} : undefined,
  }
}

const DEFAULT_QUEUE_OPTS: Partial<QueueOptions> = {
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 1000 },
    removeOnComplete: { count: 1000 },
    removeOnFail: { count: 5000 },
  },
}

export function createBrainQueues(): Record<string, Queue> {
  const connection = getRedisConnection()
  const queues: Record<string, Queue> = {}
  for (const [key, name] of Object.entries(QUEUE_NAMES)) {
    queues[key] = new Queue(name, { connection, ...DEFAULT_QUEUE_OPTS })
  }
  return queues
}

export function createWorker<T>(
  queueName: QueueName,
  processor: (job: { data: T; id?: string }) => Promise<void>,
  opts?: { concurrency?: number }
): Worker {
  const connection = getRedisConnection()
  return new Worker(queueName, async (job) => processor(job), {
    connection,
    concurrency: opts?.concurrency ?? 5,
  })
}

/**
 * Tier 1 Intake Clerk — High-throughput WAL processor.
 * Reads unconsolidated WAL entries, extracts entities/facts via Gemini Flash,
 * routes to domain-specific Section Librarian queues.
 *
 * Model: Gemini Flash (~$0.0001/message)
 * Latency target: <2s per batch
 * Concurrency: 10-20 workers
 */
import type { SupabaseClient } from '@supabase/supabase-js'
import type { Queue } from 'bullmq'
import { generateText } from 'ai'
import { models } from '@/lib/ai'
import { logger } from '@/lib/core/logger'
import { readWALTail } from './wal-emitter'
import type { KnowledgeLogEntry, SignalType, DomainType } from './types'
import type { LibrarianJob } from './worker-infra'
import { QUEUE_NAMES } from './worker-infra'

// Signal type to domain mapping
const SIGNAL_DOMAIN_MAP: Record<SignalType, DomainType> = {
  message: 'operational',
  invoice: 'financial',
  calendar: 'operational',
  pattern: 'behavioral',
  correction: 'operational',
  decision: 'operational',
  relationship: 'relational',
  pricing: 'financial',
  fiduciary: 'financial',
}

export function signalToDomain(signal: SignalType): DomainType {
  return SIGNAL_DOMAIN_MAP[signal] ?? 'operational'
}

interface ExtractedFact {
  entity_id: string
  entity_name: string
  fact: string
  domain: DomainType
}

export async function extractFactsFromBatch(
  entries: KnowledgeLogEntry[]
): Promise<ExtractedFact[]> {
  if (entries.length === 0) return []

  const batchContent = entries
    .map((e, i) => `[${i + 1}] (${e.signal_type}) ${e.content}`)
    .join('\n')

  const { text } = await generateText({
    model: models.fast,
    system: `You extract structured facts from messages. For each fact, output JSON lines:
{"entity_name": "...", "fact": "...", "domain": "financial|relational|operational|behavioral"}
Only output JSON lines, no other text. One fact per line.`,
    prompt: batchContent,
    maxTokens: 1000,
  })

  const facts: ExtractedFact[] = []
  for (const line of text.split('\n')) {
    const trimmed = line.trim()
    if (!trimmed.startsWith('{')) continue
    try {
      const parsed = JSON.parse(trimmed)
      if (parsed.entity_name && parsed.fact && parsed.domain) {
        facts.push({
          entity_id: '', // resolved downstream by Section Librarian
          entity_name: parsed.entity_name,
          fact: parsed.fact,
          domain: parsed.domain as DomainType,
        })
      }
    } catch { /* skip malformed lines */ }
  }
  return facts
}

export async function processWALBatch(
  supabase: SupabaseClient,
  orgId: string,
  domainQueues: Record<string, Queue>,
  opts?: { batchSize?: number }
): Promise<{ processed: number; routed: number }> {
  const batchSize = opts?.batchSize ?? 50

  // Read unconsolidated WAL entries
  const entries = await readWALTail(supabase, { org_id: orgId, limit: batchSize })
  if (entries.length === 0) return { processed: 0, routed: 0 }

  logger.info('intake-clerk', `Processing ${entries.length} WAL entries`, { orgId })

  // Extract facts via Gemini Flash
  const facts = await extractFactsFromBatch(entries)

  // Group facts by domain and route to Section Librarian queues
  const byDomain = new Map<DomainType, ExtractedFact[]>()
  for (const fact of facts) {
    const existing = byDomain.get(fact.domain) ?? []
    existing.push(fact)
    byDomain.set(fact.domain, existing)
  }

  let routed = 0
  for (const [domain, domainFacts] of byDomain) {
    const queueKey = domain  // matches QUEUE_NAMES keys
    const queue = domainQueues[queueKey]
    if (!queue) {
      logger.warn('intake-clerk', `No queue for domain ${domain}`)
      continue
    }

    // Group by entity for Section Librarian
    const byEntity = new Map<string, ExtractedFact[]>()
    for (const f of domainFacts) {
      const existing = byEntity.get(f.entity_name) ?? []
      existing.push(f)
      byEntity.set(f.entity_name, existing)
    }

    for (const [entityName, entityFacts] of byEntity) {
      await queue.add(`librarian-${orgId}-${entityName}`, {
        org_id: orgId,
        entity_id: '', // resolved by librarian via entity name lookup
        fact_ids: entries.filter(e => entityFacts.some(f => e.content.includes(f.fact.substring(0, 20)))).map(e => e.id),
        domain,
        entity_name: entityName,
        facts: entityFacts.map(f => f.fact),
      } satisfies LibrarianJob & { entity_name: string; facts: string[] })
      routed++
    }
  }

  // Mark WAL entries as consolidated
  const entryIds = entries.map(e => e.id)
  await supabase
    .from('knowledge_log')
    .update({ consolidated_at: new Date().toISOString() })
    .in('id', entryIds)

  logger.info('intake-clerk', `Processed ${entries.length} entries, routed ${routed} jobs`, { orgId })
  return { processed: entries.length, routed }
}
