```typescript
/**
 * System 1/2 Query Gate — Dual process query complexity classification.
 * 
 * System 1 (fast path): Dossier-only, <50ms. Simple lookups, greetings, approvals.
 * System 2 (slow path): Dossier + L3 retrieval, <500ms. Multi-entity, temporal, reasoning.
 * 
 * Inspired by VoiceAgentRAG (Salesforce, 316x speedup).
 */
import { logger } from '@/lib/core/logger'

export type QueryComplexity = 'system1' | 'system2'

// Pattern-based fast classification (no LLM needed)
export const SYSTEM1_PATTERNS = [
  /^(hi|hello|hey|good\s+(morning|afternoon|evening)|thanks|thank you)/i,
  /^(yes|no|ok|okay|approved?|reject(ed)?|cancel)/i,
  /^(what'?s|what\s+is)\s+\w+['']?s?\s+(email|phone|address|number)/i,
  /^remind\s+me/i,
  /^(send|forward|reply)\s+(it|that|this)/i,
]

export const SYSTEM2_SIGNALS = [
  /\b(why|should|analyze|compare|evaluate|assess)\b/i,
  /\b(history|timeline|full\s+story|everything\s+about)\b/i,
  /\b(before|after|last\s+(week|month|year)|since|during)\b/i,
  /\b(capacity|budget|afford|priority|trade.?off)\b/i,
  /\b(all|every|each)\s+(client|project|entity|contact)/i,
]

export function classifyQueryComplexity(
  query: string,
  opts?: { entityMentionCount?: number; conversationTurnCount?: number }
): QueryComplexity {
  const entityCount = opts?.entityMentionCount ?? 0

  // Multi-entity queries always System 2
  if (entityCount >= 3) {
    logger.debug('query-gate', 'System 2: multi-entity', { entityCount })
    return 'system2'
  }

  // Check System 1 patterns first (cheapest check)
  for (const pattern of SYSTEM1_PATTERNS) {
    if (pattern.test(query)) {
      logger.debug('query-gate', 'System 1: pattern match', { query: query.substring(0, 50) })
      return 'system1'
    }
  }

  // Check System 2 signals
  let system2Signals = 0
  for (const pattern of SYSTEM2_SIGNALS) {
    if (pattern.test(query)) system2Signals++
  }

  if (system2Signals >= 1) {
    logger.debug('query-gate', 'System 2: complexity signals', { system2Signals })
    return 'system2'
  }

  // Short queries without complexity signals -> System 1
  const wordCount = query.trim().split(/\s+/).length
  if (wordCount <= 10) {
    return 'system1'
  }

  // Default: longer queries without clear signals -> System 2 (safe)
  return 'system2'
}

/**
 * Confidence-based escalation: if System 1 assembly produces low-confidence
 * result, escalate to System 2.
 */
export function shouldEscalateToSystem2(
  system1Confidence: number,
  threshold: number = 0.6
): boolean {
  return system1Confidence < threshold
}
```
