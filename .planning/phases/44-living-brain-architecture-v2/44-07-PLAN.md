```typescript
/**
 * Predictive Coding Engine — Free Energy Principle implementation.
 * Only encodes prediction errors (surprises). Expected information
 * increments corroboration without creating new memories.
 *
 * Schema: per-entity JSONB representing expected behavior patterns.
 * Example: { payment_behavior: "pays within 14 days", communication_style: "formal",
 *            project_type: "WordPress builds, $5-8K" }
 */
import { generateText } from 'ai'
import { models } from '@/lib/ai'
import { logger } from '@/lib/core/logger'
import type { SurpriseScore, KnowledgeLogEntry, EntityDossier } from './types'

export const SURPRISE_THRESHOLD = 0.3  // Below this = expected, skip memory creation
export const SCHEMA_UPDATE_THRESHOLD = 3  // Errors needed before schema update

export async function scoreSurprise(
  fact: KnowledgeLogEntry,
  schema: Record<string, unknown>
): Promise<SurpriseScore> {
  // Empty schema — everything is novel
  if (!schema || Object.keys(schema).length === 0) {
    return { fact_id: fact.id, score: 0.5, deviation_type: 'novel_dimension' }
  }

  const schemaStr = JSON.stringify(schema, null, 2)

  const { text } = await generateText({
    model: models.fast, // Gemini Flash for fast scoring
    system: `You are a prediction error detector. Compare a new fact against an entity's behavioral schema.
Output EXACTLY one JSON line with no other text:
{"score": 0.0-1.0, "deviation_type": "expected|contradicts_schema|novel_dimension|magnitude_shift", "reason": "brief explanation"}

Scoring guide:
- 0.0: Fact perfectly matches schema expectations
- 0.1-0.29: Minor variation within expected range
- 0.3-0.5: New information not covered by schema (novel_dimension)
- 0.5-0.7: Unexpected magnitude or timing (magnitude_shift)
- 0.7-1.0: Directly contradicts schema expectation (contradicts_schema)`,
    prompt: `SCHEMA:\n${schemaStr}\n\nNEW FACT:\n${fact.content}`,
    maxTokens: 150,
  })

  try {
    const parsed = JSON.parse(text.trim())
    return {
      fact_id: fact.id,
      score: Math.max(0, Math.min(1, parsed.score ?? 0.5)),
      deviation_type: parsed.deviation_type ?? 'novel_dimension',
    }
  } catch {
    logger.warn('predictive-coding', 'Failed to parse surprise score', { text })
    return { fact_id: fact.id, score: 0.5, deviation_type: 'novel_dimension' }
  }
}

export function shouldUpdateSchema(
  errors: SurpriseScore[],
  threshold: number = SCHEMA_UPDATE_THRESHOLD
): boolean {
  // Count non-expected errors
  const significantErrors = errors.filter(e => e.score >= SURPRISE_THRESHOLD)
  // Group by deviation_type
  const byType = new Map<string, number>()
  for (const e of significantErrors) {
    byType.set(e.deviation_type, (byType.get(e.deviation_type) ?? 0) + 1)
  }
  // Update if any single type exceeds threshold
  return Array.from(byType.values()).some(count => count >= threshold)
}

export async function updateSchemaFromErrors(
  currentSchema: Record<string, unknown>,
  errors: SurpriseScore[],
  facts: KnowledgeLogEntry[]
): Promise<Record<string, unknown>> {
  const significantErrors = errors.filter(e => e.score >= SURPRISE_THRESHOLD)
  if (significantErrors.length === 0) return currentSchema

  const errorFacts = significantErrors
    .map(e => {
      const fact = facts.find(f => f.id === e.fact_id)
      return fact ? `- [${e.deviation_type}, score=${e.score.toFixed(2)}] ${fact.content}` : null
    })
    .filter(Boolean)
    .join('\n')

  const { text } = await generateText({
    model: models.fast,
    system: `You update an entity's behavioral schema based on accumulated prediction errors.
The schema is a JSON object where keys are behavioral dimensions and values are expected patterns.
Output ONLY the updated JSON schema. Merge new patterns, update contradicted patterns.`,
    prompt: `CURRENT SCHEMA:\n${JSON.stringify(currentSchema, null, 2)}\n\nPREDICTION ERRORS:\n${errorFacts}\n\nUpdated schema:`,
    maxTokens: 500,
  })

  try {
    return JSON.parse(text.trim())
  } catch {
    logger.warn('predictive-coding', 'Failed to parse updated schema', { text })
    return currentSchema
  }
}
```
