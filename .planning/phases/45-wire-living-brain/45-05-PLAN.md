   ```typescript
   /**
    * Surface high-surprise facts as proactive messages to the user.
    * Called from brain consolidation when ENABLE_PREDICTIVE_CODING=true.
    */
   export interface SurpriseEvent {
     entityName: string
     factText: string
     surpriseScore: number
     domain: DomainType
     explanation: string  // e.g. "Usually pays day 15, this invoice is day 30"
   }

   const USER_SURPRISE_THRESHOLD = 0.7

   export async function surfaceSurprises(
     supabase: SupabaseClient,
     orgId: string,
     surpriseEvents: SurpriseEvent[],
   ): Promise<{ surfaced: number }> {
     // Filter to user-facing threshold
     const highSurprise = surpriseEvents.filter(e => e.surpriseScore >= USER_SURPRISE_THRESHOLD)
     if (highSurprise.length === 0) return { surfaced: 0 }

     // Format as proactive message
     // Store in a proactive_alerts table or queue for the next conversation turn
     // The message should be natural: "Heads up: Alice's payment is 15 days later than usual"
     // DO NOT send via external channel (WhatsApp/iMessage) — surface in next chat turn
   }
   ```
