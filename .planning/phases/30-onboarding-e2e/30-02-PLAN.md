// e2e/empty-states.spec.ts
import { test, expect } from '@playwright/test'

test.describe('empty state rendering', () => {
  test('leads tab shows contextual empty state when no leads exist', async ({ page }) => {
    // Mock API to return empty leads
    await page.route('**/api/leads**', async (route) => {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({ leads: [] }),
      })
    })

    // Navigate to dashboard leads tab
    // ... (use established auth + navigation pattern)

    // Verify EmptyState renders
    const emptyState = page.locator('[role="status"]')
    await expect(emptyState).toBeVisible()
    await expect(emptyState).toContainText('No leads yet')
    await expect(emptyState).toContainText('BitBit captures leads')
  })
})
