# Plan 41-02: Stateful Workspace Execution + Supabase Schema

<objective>
Create the Supabase schema for workspace sessions and artifacts, implement the workspace store that manages session state, and verify that workspaces maintain state across multiple exec calls (package install persists, files persist). After this plan, workspace_exec supports stateful multi-step execution.
</objective>

<must_haves>
- Supabase migration for `workspace_sessions` and `workspace_artifacts` tables
- Workspace store module for CRUD operations on sessions
- Stateful execution: packages installed in call 1 available in call 2
- Dynamic tool compilation: agent can install deps and use them in subsequent commands
- E2B_API_KEY added to env validation
</must_haves>

```sql
-- Ephemeral workspace sessions (Phase 41: WKSP-01 through WKSP-08)
create table if not exists workspace_sessions (
  id uuid primary key default gen_random_uuid(),
  org_id uuid not null references organizations(id) on delete cascade,
  task_id uuid, -- FK to async_tasks if dispatched from Phase 39 engine
  sandbox_id text not null,
  status text not null default 'pending'
    check (status in ('pending', 'running', 'completed', 'failed', 'timeout')),
  purpose text not null,
  template text not null default 'default'
    check (template in ('default', 'data-science', 'web-dev')),
  started_at timestamptz not null default now(),
  completed_at timestamptz,
  total_seconds numeric not null default 0,
  cost_usd numeric not null default 0,
  created_at timestamptz not null default now()
);

-- RLS: org-scoped access
alter table workspace_sessions enable row level security;

create policy "workspace_sessions_org_read" on workspace_sessions
  for select using (org_id = (auth.jwt() ->> 'org_id')::uuid);

create policy "workspace_sessions_org_insert" on workspace_sessions
  for insert with check (org_id = (auth.jwt() ->> 'org_id')::uuid);

create policy "workspace_sessions_org_update" on workspace_sessions
  for update using (org_id = (auth.jwt() ->> 'org_id')::uuid);

-- Index for active session lookups
create index idx_workspace_sessions_org_status on workspace_sessions(org_id, status);
create index idx_workspace_sessions_sandbox on workspace_sessions(sandbox_id);

-- Workspace artifacts (outputs from workspace execution)
create table if not exists workspace_artifacts (
  id uuid primary key default gen_random_uuid(),
  workspace_id uuid not null references workspace_sessions(id) on delete cascade,
  artifact_type text not null
    check (artifact_type in ('code', 'file', 'image', 'data_table', 'log')),
  name text not null,
  content text, -- inline content for small artifacts (< 100KB)
  storage_path text, -- Supabase Storage path for large artifacts
  mime_type text,
  size_bytes integer,
  created_at timestamptz not null default now()
);

-- RLS via workspace_sessions join
alter table workspace_artifacts enable row level security;

create policy "workspace_artifacts_org_read" on workspace_artifacts
  for select using (
    exists (
      select 1 from workspace_sessions ws
      where ws.id = workspace_artifacts.workspace_id
      and ws.org_id = (auth.jwt() ->> 'org_id')::uuid
    )
  );

create policy "workspace_artifacts_org_insert" on workspace_artifacts
  for insert with check (
    exists (
      select 1 from workspace_sessions ws
      where ws.id = workspace_artifacts.workspace_id
      and ws.org_id = (auth.jwt() ->> 'org_id')::uuid
    )
  );

create index idx_workspace_artifacts_workspace on workspace_artifacts(workspace_id);
```

```typescript
import type { SupabaseClient } from '@supabase/supabase-js'
import type { WorkspaceSession, WorkspaceArtifact, WorkspaceStatus, WorkspaceTemplate } from './types'
import { logger } from '@/lib/core/logger'

export async function createWorkspaceSession(
  supabase: SupabaseClient,
  orgId: string,
  opts: {
    sandboxId: string
    purpose: string
    template?: WorkspaceTemplate
    taskId?: string
  },
): Promise<WorkspaceSession> {
  const { data, error } = await supabase
    .from('workspace_sessions')
    .insert({
      org_id: orgId,
      sandbox_id: opts.sandboxId,
      purpose: opts.purpose,
      template: opts.template || 'default',
      task_id: opts.taskId || null,
      status: 'running',
    })
    .select()
    .single()

  if (error) throw new Error(`Failed to create workspace session: ${error.message}`)
  return mapRow(data)
}

export async function getWorkspaceSession(
  supabase: SupabaseClient,
  workspaceId: string,
): Promise<WorkspaceSession | null> {
  const { data, error } = await supabase
    .from('workspace_sessions')
    .select()
    .eq('id', workspaceId)
    .single()

  if (error) {
    if (error.code === 'PGRST116') return null
    throw new Error(`Failed to get workspace session: ${error.message}`)
  }
  return data ? mapRow(data) : null
}

export async function getActiveWorkspace(
  supabase: SupabaseClient,
  orgId: string,
): Promise<WorkspaceSession | null> {
  const { data, error } = await supabase
    .from('workspace_sessions')
    .select()
    .eq('org_id', orgId)
    .eq('status', 'running')
    .order('started_at', { ascending: false })
    .limit(1)
    .maybeSingle()

  if (error) throw new Error(`Failed to get active workspace: ${error.message}`)
  return data ? mapRow(data) : null
}

export async function updateWorkspaceStatus(
  supabase: SupabaseClient,
  workspaceId: string,
  status: WorkspaceStatus,
  extras?: { completedAt?: string; totalSeconds?: number; costUsd?: number },
): Promise<void> {
  const update: Record<string, unknown> = { status }
  if (extras?.completedAt) update.completed_at = extras.completedAt
  if (extras?.totalSeconds !== undefined) update.total_seconds = extras.totalSeconds
  if (extras?.costUsd !== undefined) update.cost_usd = extras.costUsd

  const { error } = await supabase
    .from('workspace_sessions')
    .update(update)
    .eq('id', workspaceId)

  if (error) throw new Error(`Failed to update workspace status: ${error.message}`)
}

export async function saveWorkspaceArtifact(
  supabase: SupabaseClient,
  workspaceId: string,
  artifact: Omit<WorkspaceArtifact, 'id'>,
): Promise<string> {
  const { data, error } = await supabase
    .from('workspace_artifacts')
    .insert({
      workspace_id: workspaceId,
      artifact_type: artifact.type,
      name: artifact.name,
      content: artifact.content,
      storage_path: artifact.storagePath,
      mime_type: artifact.mimeType,
      size_bytes: artifact.sizeBytes,
    })
    .select('id')
    .single()

  if (error) throw new Error(`Failed to save workspace artifact: ${error.message}`)
  return data.id
}

export async function getDailyWorkspaceCost(
  supabase: SupabaseClient,
  orgId: string,
): Promise<number> {
  const dayAgo = new Date(Date.now() - 86400000).toISOString()
  const { data, error } = await supabase
    .from('workspace_sessions')
    .select('cost_usd')
    .eq('org_id', orgId)
    .gte('started_at', dayAgo)

  if (error) throw new Error(`Failed to get daily workspace cost: ${error.message}`)
  return (data || []).reduce((sum, r) => sum + Number(r.cost_usd || 0), 0)
}

function mapRow(row: Record<string, unknown>): WorkspaceSession {
  return {
    id: row.id as string,
    orgId: row.org_id as string,
    taskId: row.task_id as string | undefined,
    sandboxId: row.sandbox_id as string,
    status: row.status as WorkspaceStatus,
    purpose: row.purpose as string,
    template: row.template as WorkspaceTemplate,
    startedAt: row.started_at as string,
    completedAt: row.completed_at as string | undefined,
    totalSeconds: Number(row.total_seconds || 0),
    costUsd: Number(row.cost_usd || 0),
  }
}
```
