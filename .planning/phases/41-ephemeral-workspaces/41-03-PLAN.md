# Plan 41-03: Workspace Lifecycle + Output Delivery

<objective>
Implement workspace lifecycle management (auto-destroy on completion, timeout, and cost limit) and the output delivery pipeline that returns workspace results to the TAOR loop and stores them as execution evidence. After this plan, workspaces self-clean and their outputs surface in the agent conversation.
</objective>

<must_haves>
- Auto-destroy on task completion (explicit destroy call)
- Auto-destroy on configurable timeout (E2B native + application fallback)
- Cost circuit breaker: kill workspace if daily org limit exceeded
- Output truncation for chat context (12K char limit, matching MAX_TOOL_RESULT_CHARS)
- Full output stored in Supabase Storage with signed URL
- Workspace results returned to TAOR loop as structured tool results
- Orphan detection: cron-compatible sweep for stale 'running' sessions
</must_haves>

```typescript
import type { SupabaseClient } from '@supabase/supabase-js'
import type { WorkspaceProvider } from './types'
import { updateWorkspaceStatus, getDailyWorkspaceCost, getWorkspaceSession } from './workspace-store'
import { logger } from '@/lib/core/logger'

/** Default daily cost limit per org in USD */
const DEFAULT_DAILY_LIMIT_USD = 2.0

/** Stale session threshold in milliseconds (30 minutes) */
const ORPHAN_THRESHOLD_MS = 30 * 60 * 1000

export interface LifecycleConfig {
  dailyLimitUsd?: number
}

/**
 * Check if the org has budget remaining for workspace usage.
 * Returns { allowed, usedToday, limit }.
 */
export async function checkWorkspaceBudget(
  supabase: SupabaseClient,
  orgId: string,
  config?: LifecycleConfig,
): Promise<{ allowed: boolean; usedToday: number; limit: number }> {
  const limit = config?.dailyLimitUsd ?? DEFAULT_DAILY_LIMIT_USD
  const usedToday = await getDailyWorkspaceCost(supabase, orgId)
  return { allowed: usedToday < limit, usedToday, limit }
}

/**
 * Complete a workspace session: destroy sandbox, update status, record cost.
 */
export async function completeWorkspace(
  supabase: SupabaseClient,
  provider: WorkspaceProvider,
  workspaceId: string,
  status: 'completed' | 'failed' | 'timeout' = 'completed',
): Promise<void> {
  const session = await getWorkspaceSession(supabase, workspaceId)
  if (!session) {
    logger.warn('[workspace:lifecycle] Session not found for completion', { workspaceId })
    return
  }

  // Calculate duration and estimated cost
  const durationMs = Date.now() - new Date(session.startedAt).getTime()
  const totalSeconds = Math.ceil(durationMs / 1000)
  const costUsd = totalSeconds * 0.0001 // ~$0.36/hour at E2B base rate

  // Destroy the sandbox (best effort -- may already be gone)
  try {
    await provider.destroy(session.sandboxId)
  } catch (err) {
    logger.warn('[workspace:lifecycle] Sandbox destroy failed (may be already gone)', {
      workspaceId,
      sandboxId: session.sandboxId,
      error: err instanceof Error ? err.message : String(err),
    })
  }

  // Update Supabase record
  await updateWorkspaceStatus(supabase, workspaceId, status, {
    completedAt: new Date().toISOString(),
    totalSeconds,
    costUsd,
  })

  logger.info('[workspace:lifecycle] Workspace completed', {
    workspaceId,
    status,
    totalSeconds,
    costUsd,
  })
}

/**
 * Sweep orphaned workspace sessions -- sessions stuck in 'running' beyond threshold.
 * Designed to be called from a cron job or periodic health check.
 */
export async function sweepOrphanedWorkspaces(
  supabase: SupabaseClient,
  provider: WorkspaceProvider,
): Promise<{ swept: number }> {
  const cutoff = new Date(Date.now() - ORPHAN_THRESHOLD_MS).toISOString()

  const { data: orphans, error } = await supabase
    .from('workspace_sessions')
    .select('id, sandbox_id')
    .eq('status', 'running')
    .lt('started_at', cutoff)

  if (error) {
    logger.error('[workspace:lifecycle] Failed to query orphaned sessions', { error: error.message })
    return { swept: 0 }
  }

  let swept = 0
  for (const orphan of orphans || []) {
    try {
      await provider.destroy(orphan.sandbox_id)
    } catch {
      // Sandbox may already be gone
    }
    await updateWorkspaceStatus(supabase, orphan.id, 'timeout', {
      completedAt: new Date().toISOString(),
    })
    swept++
  }

  if (swept > 0) {
    logger.info('[workspace:lifecycle] Swept orphaned workspaces', { swept })
  }

  return { swept }
}
```

```typescript
import type { SupabaseClient } from '@supabase/supabase-js'
import type { WorkspaceExecResult, WorkspaceArtifact } from './types'
import { saveWorkspaceArtifact } from './workspace-store'
import { logger } from '@/lib/core/logger'

/** Maximum inline output chars (matches tool-executor MAX_TOOL_RESULT_CHARS) */
const MAX_INLINE_CHARS = 12_000

/** Maximum artifact content size for inline storage (100KB) */
const MAX_INLINE_ARTIFACT_BYTES = 100 * 1024

export interface DeliveredOutput {
  /** Truncated output safe for TAOR loop context */
  inline: string
  /** Artifact IDs stored in Supabase */
  artifactIds: string[]
  /** Whether output was truncated */
  truncated: boolean
  /** Full output size in bytes */
  fullSizeBytes: number
}

/**
 * Process workspace execution results for delivery back to the TAOR loop.
 *
 * 1. Format stdout/stderr/error into a readable string
 * 2. Truncate to MAX_INLINE_CHARS for chat context
 * 3. Store full output as artifact if truncated
 * 4. Store any generated artifacts (files, images, etc.)
 * 5. Return structured result for the tool_result message
 */
export async function deliverWorkspaceOutput(
  supabase: SupabaseClient,
  workspaceId: string,
  execResult: WorkspaceExecResult,
): Promise<DeliveredOutput> {
  const artifactIds: string[] = []

  // Format the execution output
  const sections: string[] = []
  if (execResult.error) {
    sections.push(`Error: ${execResult.error}`)
  }
  if (execResult.stdout.trim()) {
    sections.push(`Output:\n${execResult.stdout}`)
  }
  if (execResult.stderr.trim()) {
    sections.push(`Stderr:\n${execResult.stderr}`)
  }

  const fullOutput = sections.join('\n\n') || '(no output)'
  const fullSizeBytes = Buffer.byteLength(fullOutput, 'utf-8')
  let truncated = false

  // Truncate for inline delivery
  let inline = fullOutput
  if (inline.length > MAX_INLINE_CHARS) {
    inline = inline.slice(0, MAX_INLINE_CHARS) +
      '\n\n[Output truncated -- ' +
      (fullOutput.length - MAX_INLINE_CHARS).toLocaleString() +
      ' chars omitted. Full output stored as artifact.]'
    truncated = true

    // Store full output as artifact
    try {
      const id = await saveWorkspaceArtifact(supabase, workspaceId, {
        type: 'log',
        name: 'full-output.txt',
        content: fullSizeBytes <= MAX_INLINE_ARTIFACT_BYTES ? fullOutput : undefined,
        storagePath: fullSizeBytes > MAX_INLINE_ARTIFACT_BYTES
          ? await storeInStorage(supabase, workspaceId, 'full-output.txt', fullOutput)
          : undefined,
        mimeType: 'text/plain',
        sizeBytes: fullSizeBytes,
      })
      artifactIds.push(id)
    } catch (err) {
      logger.warn('[workspace:output] Failed to store full output artifact', {
        workspaceId,
        error: err instanceof Error ? err.message : String(err),
      })
    }
  }

  // Store any execution artifacts (files, images, etc.)
  if (execResult.artifacts) {
    for (const artifact of execResult.artifacts) {
      try {
        const id = await saveWorkspaceArtifact(supabase, workspaceId, artifact)
        artifactIds.push(id)
      } catch (err) {
        logger.warn('[workspace:output] Failed to store artifact', {
          workspaceId,
          artifactName: artifact.name,
          error: err instanceof Error ? err.message : String(err),
        })
      }
    }
  }

  return { inline, artifactIds, truncated, fullSizeBytes }
}

/**
 * Store large content in Supabase Storage and return the path.
 */
async function storeInStorage(
  supabase: SupabaseClient,
  workspaceId: string,
  filename: string,
  content: string,
): Promise<string> {
  const path = `workspaces/${workspaceId}/${filename}`

  const { error } = await supabase.storage
    .from('workspace-outputs')
    .upload(path, Buffer.from(content, 'utf-8'), {
      contentType: 'text/plain',
      upsert: true,
    })

  if (error) throw new Error(`Storage upload failed: ${error.message}`)
  return path
}
```
